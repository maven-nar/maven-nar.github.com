25   * Test for Microsoft Developer Studio linker
26   *
27   * Override create to test concrete compiler implementions

29  public class TestInstalledDevStudioLinker extends TestDevStudioLinker
30  {
31       public TestInstalledDevStudioLinker(String name) {
32          super(name);
33       }
34  
35      public void failingtestGetLibraryPath() {
36          File[] libpath = DevStudioLinker.getInstance().getLibraryPath();
37          //
38          //  unless you tweak the library path
39          //       it should have more thean three entries
40          assertTrue(libpath.length >= 2);
41          //
42          //   check if these files can be found         
43          //
44          String[] libnames = new String[] { "kernel32.lib", 
45              "advapi32.lib", "msvcrt.lib", "mfc42.lib", "mfc70.lib" };
46          boolean[] libfound = new boolean[libnames.length];
47          for (int i = 0; i < libpath.length; i++) {
48             for (int j = 0; j < libnames.length; j++) {
49                 File libfile = new File(libpath[i], libnames[j]);
50                 if (libfile.exists()) {
51                    libfound[j] = true;
52                 }
53             }
54          }
55          assertTrue("kernel32 not found", libfound[0]);
56          assertTrue("advapi32 not found", libfound[1]);
57          assertTrue("msvcrt not found", libfound[2]);
58          if(!(libfound[3] || libfound[4])) {
59             fail("mfc42.lib or mfc70.lib not found");
60          }
61      }
62  }
