122     public void testAddLibrarySetDirSwitch() {
123         AbstractLdLinker linker = getLinker();
124         CCTask task = new CCTask();
125         LibrarySet[] sets = new LibrarySet[]{new LibrarySet()};
126         /* throws an Exception in setLibs otherwise */
127         sets[0].setProject(new org.apache.tools.ant.Project());
128         sets[0].setDir(new File("/foo"));
129         sets[0].setLibs(new CUtil.StringArrayBuilder("bart,cart,dart"));
130         Vector preargs = new Vector();
131         Vector midargs = new Vector();
132         Vector endargs = new Vector();
133         String[] rc = linker.addLibrarySets(task, sets, preargs, midargs,
134                 endargs);
135         String libdirSwitch = (String) endargs.elementAt(0);
136         assertEquals(libdirSwitch.substring(0, 2), "-L");
137         //
138         //  can't have space after -L or will break Mac OS X
139         //
140         assertTrue(!libdirSwitch.substring(2, 3).equals(" "));
141         assertEquals(libdirSwitch.substring(libdirSwitch.length() - 3), "foo");
142     }

243                 endargs);

248             assertEquals("-lbart", (String) endargs.elementAt(0));
249             assertEquals("-lcart", (String) endargs.elementAt(1));
250             assertEquals("-ldart", (String) endargs.elementAt(2));
