65    /**
66     * This method tests CompilerDef.getActiveDefines.
67     *
68     * A CompilerDef is created similar to what would be created for
69     *
70     * <cc><defineset><define name="DEBUG" if="debug"/> <define name="NDEBUG"
71     * unless="debug"/> </defineset> </cc>
72     *
73     * Then getActiveDefines is called for a project without and with the
74     * "debug" property defined. Return value from getActiveDefines should
75     * contain one member
76     */
77    public void testGetActiveDefines() {
78      Project project = new org.apache.tools.ant.Project();
79      CompilerDef def = new CompilerDef();
80      def.setProject(project);
81      DefineSet defset = new DefineSet();
82      DefineArgument arg1 = new DefineArgument();
83      arg1.setName("DEBUG");
84      arg1.setIf("debug");
85      defset.addDefine(arg1);
86      DefineArgument arg2 = new DefineArgument();
87      arg2.setName("NDEBUG");
88      arg2.setUnless("debug");
89      defset.addDefine(arg2);
90      def.addConfiguredDefineset(defset);
91      //
92      //  Evaluate without "debug" set
93      //
94      UndefineArgument[] activeArgs = def.getActiveDefines();
95      assertEquals(1, activeArgs.length);
96      assertEquals("NDEBUG", activeArgs[0].getName());
97      //
98      //  Set the "debug" property
99      //
100     project.setProperty("debug", "");
101     activeArgs = def.getActiveDefines();
102     assertEquals(1, activeArgs.length);
103     assertEquals("DEBUG", activeArgs[0].getName());
104   }

130     project.setProperty("debug", "");
131     includePaths = def.getActiveIncludePaths();

190   /**
191    * Gets the command line arguments that precede filenames.
192    *
193    * This method filters out <i>-m</i> options because they are
194    * platform-specific options of the GNU compiler suite.
195    *
196    * @param processor
197    *            processor under test
198    * @return command line arguments
199    */
200   protected String[] getPreArguments(final ProcessorDef processor) {
201     final String[] result =
202       ((CommandLineCompilerConfiguration) getConfiguration(processor))
203       .getPreArguments();
204 
205     // filter out -m (i.e. platform-specific) options
206     int j = -1;
207     for (int i = 0; i < result.length; i++) {
208       if (result[i].startsWith("-m")) continue;
209       if (i != ++j) result[j] = result[i];
210     }
211     if (++j == result.length) return result;
212     final String[] filtered = new String[j];
213     System.arraycopy(result, 0, filtered, 0, j);
214     return filtered;
215   }
