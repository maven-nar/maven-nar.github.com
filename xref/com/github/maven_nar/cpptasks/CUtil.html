47      public static final class StringArrayBuilder {
48          private String[] _value;
49          public StringArrayBuilder(String value) {
50              // Split the defines up
51              StringTokenizer tokens = new StringTokenizer(value, ", ");
52              Vector vallist = new Vector();
53              while (tokens.hasMoreTokens()) {
54                  String val = tokens.nextToken().trim();
55                  if (val.length() == 0) {
56                      continue;
57                  }
58                  vallist.addElement(val);
59              }
60              _value = new String[vallist.size()];
61              vallist.copyInto(_value);
62          }
63          public String[] getValue() {
64              return _value;
65          }
66      }

79       * Checks a array of names for non existent or non directory entries and
80       * nulls them out.
81       *

84      public static int checkDirectoryArray(String[] names) {
85          int count = 0;
86          for (int i = 0; i < names.length; i++) {
87              if (names[i] != null) {
88                  File dir = new File(names[i]);
89                  if (dir.exists() && dir.isDirectory()) {
90                      count++;
91                  } else {
92                      names[i] = null;
93                  }
94              }
95          }
96          return count;
97      }

111     /**
112      * Gets the parent directory for the executable file name using the current
113      * directory and system executable path
114      *
115      * @param exeName
116      *            Name of executable such as "cl.exe"
117      * @return parent directory or null if not located

119     public static File getExecutableLocation(String exeName) {
120         //
121         //   must add current working directory to the
122         //      from of the path from the "path" environment variable
123         File currentDir = new File(System.getProperty("user.dir"));
124         if (new File(currentDir, exeName).exists()) {
125             return currentDir;
126         }
127         File[] envPath = CUtil.getPathFromEnvironment("PATH",
128                 File.pathSeparator);
129         for (int i = 0; i < envPath.length; i++) {
130             if (new File(envPath[i], exeName).exists()) {
131                 return envPath[i];
132             }
133         }
134         return null;
135     }

146     /**
147      * Returns an array of File for each existing directory in the specified
148      * environment variable
149      *
150      * @param envVariable
151      *            environment variable name such as "LIB" or "INCLUDE"
152      * @param delim
153      *            delimitor used to separate parts of the path, typically ";"
154      *            or ":"
155      * @return array of File's for each part that is an existing directory
156      */
157     public static File[] getPathFromEnvironment(String envVariable, String delim) {
158         // OS/4000 does not support the env command.
159         if (System.getProperty("os.name").equals("OS/400"))
160             return new File[]{};
161         Vector osEnv = Execute.getProcEnvironment();
162         String match = envVariable.concat("=");
163         for (Enumeration e = osEnv.elements(); e.hasMoreElements();) {
164             String entry = ((String) e.nextElement()).trim();
165             if (entry.length() > match.length()) {
166                 String entryFrag = entry.substring(0, match.length());
167                 if (entryFrag.equalsIgnoreCase(match)) {
168                     String path = entry.substring(match.length());
169                     return parsePath(path, delim);
170                 }
171             }
172         }
173         File[] noPath = new File[0];
174         return noPath;
175     }
176     /**
177      * Returns a relative path for the targetFile relative to the base
178      * directory.
179      *
180      * @param base
181      *            base directory as returned by File.getCanonicalPath()
182      * @param targetFile
183      *            target file
184      * @return relative path of target file. Returns targetFile if there were
185      *         no commonalities between the base and the target
186      *
187      */
188     public static String getRelativePath(final String base, final File targetFile) {
189         try {
190             //
191             //   remove trailing file separator
192             //
193             String canonicalBase = base;
194             if (base.charAt(base.length() - 1) != File.separatorChar) {
195                 canonicalBase = base + File.separatorChar;
196             }
197             //
198             //   get canonical name of target
199             //
200             String canonicalTarget;
201             if (System.getProperty("os.name").equals("OS/400"))
202                 canonicalTarget = targetFile.getPath();
203             else
204                 canonicalTarget = targetFile.getCanonicalPath();
205             if (canonicalBase.startsWith(canonicalTarget + File.separatorChar)) {
206                 canonicalTarget = canonicalTarget + File.separator;
207             }
208             if (canonicalTarget.equals(canonicalBase)) {
209                 return ".";
210             }
211             //
212             //  see if the prefixes are the same
213             //
214             if (canonicalBase.substring(0, 2).equals("\\\\")) {
215                 //
216                 //  UNC file name, if target file doesn't also start with same
217                 //      server name, don't go there
218                 int endPrefix = canonicalBase.indexOf('\\', 2);
219                 String prefix1 = canonicalBase.substring(0, endPrefix);
220                 String prefix2 = canonicalTarget.substring(0, endPrefix);
221                 if (!prefix1.equals(prefix2)) {
222                     return canonicalTarget;
223                 }
224             } else {
225                 if (canonicalBase.substring(1, 3).equals(":\\")) {
226                     int endPrefix = 2;
227                     String prefix1 = canonicalBase.substring(0, endPrefix);
228                     String prefix2 = canonicalTarget.substring(0, endPrefix);
229                     if (!prefix1.equals(prefix2)) {
230                         return canonicalTarget;
231                     }
232                 } else {
233                     if (canonicalBase.charAt(0) == '/') {
234                         if (canonicalTarget.charAt(0) != '/') {
235                             return canonicalTarget;
236                         }
237                     }
238                 }
239             }
240             char separator = File.separatorChar;
241             int lastCommonSeparator = -1;
242             int minLength = canonicalBase.length();
243             if (canonicalTarget.length() < minLength) {
244                 minLength = canonicalTarget.length();
245             }
246             //
247             //  walk to the shorter of the two paths
248             //      finding the last separator they have in common
249             for (int i = 0; i < minLength; i++) {
250                 if (canonicalTarget.charAt(i) == canonicalBase.charAt(i)) {
251                     if (canonicalTarget.charAt(i) == separator) {
252                         lastCommonSeparator = i;
253                     }
254                 } else {
255                     break;
256                 }
257             }
258             StringBuffer relativePath = new StringBuffer(50);
259             //
260             //   walk from the first difference to the end of the base
261             //      adding "../" for each separator encountered
262             //
263             for (int i = lastCommonSeparator + 1; i < canonicalBase.length(); i++) {
264                 if (canonicalBase.charAt(i) == separator) {
265                     if (relativePath.length() > 0) {
266                         relativePath.append(separator);
267                     }
268                     relativePath.append("..");
269                 }
270             }
271             if (canonicalTarget.length() > lastCommonSeparator + 1) {
272                 if (relativePath.length() > 0) {
273                     relativePath.append(separator);
274                 }
275                 relativePath.append(canonicalTarget.substring(lastCommonSeparator + 1));
276             }
277             return relativePath.toString();
278         } catch (IOException ex) {
279         }
280         return targetFile.toString();
281     }
282     public static boolean isActive(Project p, String ifCond, String unlessCond)
283             throws BuildException {
284         if (ifCond != null) {
285             String ifValue = p.getProperty(ifCond);
286             if (ifValue == null) {
287                 return false;
288             } else {
289                 if (ifValue.equals("false") || ifValue.equals("no")) {
290                     throw new BuildException("if condition \"" + ifCond
291                             + "\" has suspicious value \"" + ifValue);
292                 }
293             }
294         }
295         if (unlessCond != null) {
296             String unlessValue = p.getProperty(unlessCond);
297             if (unlessValue != null) {
298                 if (unlessValue.equals("false") || unlessValue.equals("no")) {
299                     throw new BuildException("unless condition \"" + unlessCond
300                             + "\" has suspicious value \"" + unlessValue);
301                 }
302                 return false;
303             }
304         }
305         return true;
306     }
307     /**
308      * Parse a string containing directories into an File[]
309      *
310      * @param path
311      *            path string, for example ".;c:\something\include"
312      * @param delim
313      *            delimiter, typically ; or :
314      */
315     public static File[] parsePath(String path, String delim) {
316         Vector libpaths = new Vector();
317         int delimPos = 0;
318         for (int startPos = 0; startPos < path.length(); startPos = delimPos
319                 + delim.length()) {
320             delimPos = path.indexOf(delim, startPos);
321             if (delimPos < 0) {
322                 delimPos = path.length();
323             }
324             //
325             //   don't add an entry for zero-length paths
326             //
327             if (delimPos > startPos) {
328                 String dirName = path.substring(startPos, delimPos);
329                 File dir = new File(dirName);
330                 if (dir.exists() && dir.isDirectory()) {
331                     libpaths.addElement(dir);
332                 }
333             }
334         }
335         File[] paths = new File[libpaths.size()];
336         libpaths.copyInto(paths);
337         return paths;
338     }

343     public static int runCommand(CCTask task, File workingDir,
344             String[] cmdline, boolean newEnvironment, Environment env)
345             throws BuildException {
346         try {
347             task.log(Commandline.toString(cmdline), task.getCommandLogLevel() );
348             Execute exe = new Execute(new LogStreamHandler(task,
349                     Project.MSG_INFO, Project.MSG_ERR));
350             if (System.getProperty("os.name").equals("OS/390"))
351                 exe.setVMLauncher(false);
352             exe.setAntRun(task.getProject());
353             exe.setCommandline(cmdline);
354             exe.setWorkingDirectory(workingDir);
355             if (env != null) {
356                 String[] environment = env.getVariables();
357                 if (environment != null) {
358                     for (int i = 0; i < environment.length; i++) {
359                         task.log("Setting environment variable: "
360                                 + environment[i], Project.MSG_VERBOSE);
361                     }
362                 }
363                 exe.setEnvironment(environment);
364             }
365             exe.setNewenvironment(newEnvironment);
366             return exe.execute();
367         } catch (java.io.IOException exc) {
368             throw new BuildException("Could not launch " + cmdline[0] + ": "
369                     + exc, task.getLocation());
370         }
371     }

401     /**
402      * Compares the contents of an array and a Vector for set equality. Assumes
403      * input array and vector are sets (i.e. no duplicate entries)
404      */
405     public static boolean sameSet(Object[] a, Vector b) {
406         if (a == null || b == null || a.length != b.size()) {
407             return false;
408         }
409         if (a.length == 0) {
410             return true;
411         }
412         // Convert the array into a set
413         Hashtable t = new Hashtable();
414         for (int i = 0; i < a.length; i++) {
415             t.put(a[i], a[i]);
416         }
417         for (int i = 0; i < b.size(); i++) {
418             Object o = b.elementAt(i);
419             if (t.remove(o) == null) {
420                 return false;
421             }
422         }
423         return (t.size() == 0);
424     }

433     /**
434      * Replaces any embedded quotes in the string so that the value can be
435      * placed in an attribute in an XML file
436      *
437      * @param attrValue
438      *            value to be expressed
439      * @return equivalent attribute literal
440      *
441      */
442     public static String xmlAttribEncode(String attrValue) {
443     	StringBuffer buf = new StringBuffer (attrValue);
444     	int quotePos;
445     	
446         for (quotePos = -1; (quotePos = buf.indexOf("\"", quotePos + 1)) >= 0;) {
447         	buf.deleteCharAt(quotePos);
448         	buf.insert (quotePos, "&quot;");
449         	quotePos += 5;
450         }
451         
452         for (quotePos = -1; (quotePos = buf.indexOf("<", quotePos + 1)) >= 0;) {
453         	buf.deleteCharAt(quotePos);
454         	buf.insert (quotePos, "&lt;");
455         	quotePos += 3;
456         }
457         
458         for (quotePos = -1; (quotePos = buf.indexOf(">", quotePos + 1)) >= 0;) {
459         	buf.deleteCharAt(quotePos);
460         	buf.insert (quotePos, "&gt;");
461         	quotePos += 3;
462         }
463         
464         return buf.toString();
465     }

467     public final static int FILETIME_EPSILON = 500;
468     /**
469      * Determines whether time1 is earlier than time2
470      * to a degree that file system time truncation is not significant.
471      *
472      * @param time1 long first time value
473      * @param time2 long second time value
474      * @return boolean if first time value is earlier than second time value.
475      * If the values are within the rounding error of the file system return false.
476      */
477     public static boolean isSignificantlyBefore(long time1, long time2) {
478       return (time1 + FILETIME_EPSILON) < time2;
479     }

487      * If the values are within the rounding error of the file system return false.
488      */

490       return time1 > (time2 + FILETIME_EPSILON);

509      *  Determines if source file has a system path,
510      *    that is part of the compiler or platform.
511      *   @param source source, may not be null.
512      *   @return true is source file appears to be system library
513      *         and its path should be discarded.
514      */
515      public static boolean isSystemPath(final File source) {
516          String lcPath = source.getAbsolutePath().toLowerCase(java.util.Locale.US);
517          return lcPath.indexOf("platformsdk") != -1
518              || lcPath.indexOf("microsoft") != -1 ||
519                 lcPath == "/usr/include" ||
520                 lcPath == "/usr/lib" ||
521                 lcPath == "/usr/local/include" ||
522                 lcPath == "/usr/local/lib";
523      }
