74      /**
75       * Adds command-line arguments for include directories.
76       * 
77       * If relativeArgs is not null will add corresponding relative paths
78       * include switches to that vector (for use in building a configuration
79       * identifier that is consistent between machines).
80       * 
81       * @param baseDirPath Base directory path.
82       * @param includeDirs
83       *            Array of include directory paths
84       * @param args
85       *            Vector of command line arguments used to execute the task
86       * @param relativeArgs
87       *            Vector of command line arguments used to build the
88       *            configuration identifier
89       */
90      protected void addIncludes(String baseDirPath, File[] includeDirs,
91              Vector<String> args, Vector<String> relativeArgs, StringBuffer includePathId,
92              boolean isSystem) {
93          for (int i = 0; i < includeDirs.length; i++) {
94              args.addElement(getIncludeDirSwitch(includeDirs[i]
95  					.getAbsolutePath(), isSystem));
96              if (relativeArgs != null) {
97  				String relative = CUtil.getRelativePath(baseDirPath,
98                          includeDirs[i]);
99  				relativeArgs.addElement(getIncludeDirSwitch(relative, isSystem));
100                 if (includePathId != null) {
101                     if (includePathId.length() == 0) {
102                         includePathId.append("/I");
103                     } else {
104                         includePathId.append(" /I");
105                     }
106                     includePathId.append(relative);
107                 }
108             }
109         }
110     }

112     protected void buildDefineArguments(CompilerDef[] defs, Vector<String> args) {
113         //
114         //   assume that we aren't inheriting defines from containing <cc>
115         //
116         UndefineArgument[] merged = defs[0].getActiveDefines();
117         for (int i = 1; i < defs.length; i++) {
118             //
119             //  if we are inheriting, merge the specific defines with the
120             //      containing defines
121             merged = UndefineArgument.merge(defs[i].getActiveDefines(), merged);
122         }
123         StringBuffer buf = new StringBuffer(30);
124         for (int i = 0; i < merged.length; i++) {
125             buf.setLength(0);
126             UndefineArgument current = merged[i];
127             if (current.isDefine()) {
128                 getDefineSwitch(buf, current.getName(), current.getValue());
129             } else {
130                 getUndefineSwitch(buf, current.getName());
131             }
132             args.addElement(buf.toString());
133         }

139     public void compile(CCTask task, File outputDir, String[] sourceFiles,
140             String[] args, String[] endArgs, boolean relentless,
141             CommandLineCompilerConfiguration config, ProgressMonitor monitor)
142             throws BuildException {
143         BuildException exc = null;
144         //
145         //   determine length of executable name and args
146         //
147         String command = getCommandWithPath(config);
148         int baseLength = command.length() + args.length + endArgs.length;
149         if (libtool) {
150             baseLength += 8;
151         }
152         for (int i = 0; i < args.length; i++) {
153             baseLength += args[i].length();
154         }
155         for (int i = 0; i < endArgs.length; i++) {
156             baseLength += endArgs[i].length();
157         }
158         if (baseLength > getMaximumCommandLength()) {
159             throw new BuildException(
160                     "Command line is over maximum length without specifying source file");
161         }
162         //
163         //  typically either 1 or Integer.MAX_VALUE
164         //
165         int maxInputFilesPerCommand = getMaximumInputFilesPerCommand();
166         int argumentCountPerInputFile = getArgumentCountPerInputFile();
167         for (int sourceIndex = 0; sourceIndex < sourceFiles.length;) {
168             int cmdLength = baseLength;
169             int firstFileNextExec;
170             for (firstFileNextExec = sourceIndex; firstFileNextExec < sourceFiles.length
171                     && (firstFileNextExec - sourceIndex) < maxInputFilesPerCommand; firstFileNextExec++) {
172                 cmdLength += getTotalArgumentLengthForInputFile(outputDir,
173                         sourceFiles[firstFileNextExec]);
174                 if (cmdLength >= getMaximumCommandLength())
175                     break;
176             }
177             if (firstFileNextExec == sourceIndex) {
178                 throw new BuildException(
179                         "Extremely long file name, can't fit on command line");
180             }
181             int argCount = args.length + 1 + endArgs.length
182                     + (firstFileNextExec - sourceIndex)
183                     * argumentCountPerInputFile;
184             if (libtool) {
185                 argCount++;
186             }
187             String[] commandline = new String[argCount];
188             int index = 0;
189             if (libtool) {
190                 commandline[index++] = "libtool";
191             }
192             commandline[index++] = command;
193             for (int j = 0; j < args.length; j++) {
194                 commandline[index++] = args[j];
195             }
196             for (int j = sourceIndex; j < firstFileNextExec; j++) {
197                 for (int k = 0; k < argumentCountPerInputFile; k++) {
198                     commandline[index++] = getInputFileArgument(outputDir,
199                             sourceFiles[j], k);
200                 }
201             }
202             for (int j = 0; j < endArgs.length; j++) {
203                 commandline[index++] = endArgs[j];
204             }
205             int retval = runCommand(task, outputDir, commandline);
206             if (monitor != null) {
207                 String[] fileNames = new String[firstFileNextExec - sourceIndex];
208 
209                 for (int j = 0; j < fileNames.length; j++) {
210                     fileNames[j] = sourceFiles[sourceIndex + j];
211                 }
212                 monitor.progress(fileNames);
213             }
214             //
215             //   if the process returned a failure code and
216             //      we aren't holding an exception from an earlier
217             //      interation
218             if (retval != 0 && exc == null) {
219                 //
220                 //   construct the exception
221                 //
222                 exc = new BuildException(getCommandWithPath(config)
223                         + " failed with return code " + retval, task
224                         .getLocation());
225 
226                 //
227                 //   and throw it now unless we are relentless
228                 //
229                 if (!relentless) {
230                     throw exc;
231                 }
232             }
233             sourceIndex = firstFileNextExec;
234         }
235         //
236         //   if the compiler returned a failure value earlier
237         //      then throw an exception
238         if (exc != null) {
239             throw exc;
240         }
241     }
242     protected CompilerConfiguration createConfiguration(final CCTask task,
243             final LinkType linkType, 
244 			final ProcessorDef[] baseDefs, 
245 			final CompilerDef specificDef,
246 			final TargetDef targetPlatform,
247 			final VersionInfo versionInfo) {
248         Vector<String> args = new Vector<String>();
249         CompilerDef[] defaultProviders = new CompilerDef[baseDefs.length + 1];
250         for (int i = 0; i < baseDefs.length; i++) {
251             defaultProviders[i + 1] = (CompilerDef) baseDefs[i];
252         }
253         defaultProviders[0] = specificDef;
254         Vector<CommandLineArgument> cmdArgs = new Vector<CommandLineArgument>();
255         //
256         //   add command line arguments inherited from <cc> element
257         //     any "extends" and finally the specific CompilerDef
258         CommandLineArgument[] commandArgs;
259         for (int i = defaultProviders.length - 1; i >= 0; i--) {
260             commandArgs = defaultProviders[i].getActiveProcessorArgs();
261             for (int j = 0; j < commandArgs.length; j++) {
262                 if (commandArgs[j].getLocation() == 0) {
263                     String arg = commandArgs[j].getValue();
264                     if (isWindows() && arg.matches(".*[ \"].*")) {
265                         // Work around inconsistent quoting by Ant
266                         arg = "\"" + arg.replaceAll("[\\\\\"]", "\\\\$0") + "\"";
267                     }
268                     args.addElement(arg);
269                 } else {
270                     cmdArgs.addElement(commandArgs[j]);
271                 }
272             }
273         }
274         Vector<ProcessorParam> params = new Vector<ProcessorParam>();
275         //
276         //   add command line arguments inherited from <cc> element
277         //     any "extends" and finally the specific CompilerDef
278         ProcessorParam[] paramArray;
279         for (int i = defaultProviders.length - 1; i >= 0; i--) {
280             paramArray = defaultProviders[i].getActiveProcessorParams();
281             for (int j = 0; j < paramArray.length; j++) {
282                 params.add(paramArray[j]);
283             }
284         }
285         paramArray = (ProcessorParam[]) (params.toArray(new ProcessorParam[params.size()]));
286 
287         if (specificDef.isClearDefaultOptions() == false) {
288             boolean multithreaded = specificDef.getMultithreaded(defaultProviders, 1);
289             boolean debug = specificDef.getDebug(baseDefs, 0);
290             boolean exceptions = specificDef.getExceptions(defaultProviders, 1);
291             Boolean rtti = specificDef.getRtti(defaultProviders, 1);
292             OptimizationEnum optimization = specificDef.getOptimization(defaultProviders, 1);
293             this.addImpliedArgs(args, debug, multithreaded, exceptions, linkType, rtti, optimization);
294         }
295 
296         //
297         //    add all appropriate defines and undefines
298         //
299         buildDefineArguments(defaultProviders, args);
300         int warnings = specificDef.getWarnings(defaultProviders, 0);
301         addWarningSwitch(args, warnings);
302         Enumeration<CommandLineArgument> argEnum = cmdArgs.elements();
303         int endCount = 0;
304         while (argEnum.hasMoreElements()) {
305             CommandLineArgument arg = argEnum.nextElement();
306             switch (arg.getLocation()) {
307                 case 1 :
308                     args.addElement(arg.getValue());
309                     break;
310                 case 2 :
311                     endCount++;
312                     break;
313             }
314         }
315         String[] endArgs = new String[endCount];
316         argEnum = cmdArgs.elements();
317         int index = 0;
318         while (argEnum.hasMoreElements()) {
319             CommandLineArgument arg = (CommandLineArgument) argEnum
320                     .nextElement();
321             if (arg.getLocation() == 2) {
322                 endArgs[index++] = arg.getValue();
323             }
324         }
325         //
326         //   Want to have distinct set of arguments with relative
327         //      path names for includes that are used to build
328         //      the configuration identifier
329         //
330         Vector<String> relativeArgs = (Vector) args.clone();
331         //
332         //    add all active include and sysincludes
333         //
334         StringBuffer includePathIdentifier = new StringBuffer();
335         File baseDir = specificDef.getProject().getBaseDir();
336         String baseDirPath;
337         try {
338             baseDirPath = baseDir.getCanonicalPath();
339         } catch (IOException ex) {
340             baseDirPath = baseDir.toString();
341         }
342         Vector<String> includePath = new Vector<String>();
343         Vector<String> sysIncludePath = new Vector<String>();
344         for (int i = defaultProviders.length - 1; i >= 0; i--) {
345             String[] incPath = defaultProviders[i].getActiveIncludePaths();
346             for (int j = 0; j < incPath.length; j++) {
347                 includePath.addElement(incPath[j]);
348             }
349             incPath = defaultProviders[i].getActiveSysIncludePaths();
350             for (int j = 0; j < incPath.length; j++) {
351                 sysIncludePath.addElement(incPath[j]);
352             }
353         }
354         File[] incPath = new File[includePath.size()];
355         for (int i = 0; i < includePath.size(); i++) {
356             incPath[i] = new File((String) includePath.elementAt(i));
357         }
358         File[] sysIncPath = new File[sysIncludePath.size()];
359         for (int i = 0; i < sysIncludePath.size(); i++) {
360             sysIncPath[i] = new File((String) sysIncludePath.elementAt(i));
361         }
362         addIncludes(baseDirPath, incPath, args, relativeArgs,
363 				includePathIdentifier, false);
364 		addIncludes(baseDirPath, sysIncPath, args, null, null, true);
365         StringBuffer buf = new StringBuffer(getIdentifier());
366         for (int i = 0; i < relativeArgs.size(); i++) {
367             buf.append(' ');
368             buf.append(relativeArgs.elementAt(i));
369         }
370         for (int i = 0; i < endArgs.length; i++) {
371             buf.append(' ');
372             buf.append(endArgs[i]);
373         }
374         String configId = buf.toString();
375         String[] argArray = new String[args.size()];
376         args.copyInto(argArray);
377         boolean rebuild = specificDef.getRebuild(baseDefs, 0);
378         File[] envIncludePath = getEnvironmentIncludePath();
379         String path = specificDef.getToolPath();
380         return new CommandLineCompilerConfiguration(this, configId, incPath,
381                 sysIncPath, envIncludePath, includePathIdentifier.toString(),
382                 argArray, paramArray, rebuild, endArgs, path);
383     }

390     public String getCommandWithPath(CommandLineCompilerConfiguration config) {
391         if( config.getCommandPath() != null ) {
392             File command = new File( config.getCommandPath(), this.getCommand() );
393             try {
394                 return command.getCanonicalPath();
395             } catch (IOException e) {
396                 e.printStackTrace();
397                 return command.getAbsolutePath();
398             }
399         }
400         else
401         {
402             return this.getCommand();
403         }
