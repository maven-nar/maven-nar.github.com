84    /**
85     * Adds command-line arguments for include directories.
86     * 
87     * If relativeArgs is not null will add corresponding relative paths
88     * include switches to that vector (for use in building a configuration
89     * identifier that is consistent between machines).
90     * 
91     * @param baseDirPath
92     *          Base directory path.
93     * @param includeDirs
94     *          Array of include directory paths
95     * @param args
96     *          Vector of command line arguments used to execute the task
97     * @param relativeArgs
98     *          Vector of command line arguments used to build the
99     *          configuration identifier
100    */
101   protected void addIncludes(final String baseDirPath, final File[] includeDirs, final Vector<String> args,
102       final Vector<String> relativeArgs, final StringBuffer includePathId, final boolean isSystem) {
103     for (final File includeDir : includeDirs) {
104       args.addElement(getIncludeDirSwitch(includeDir.getAbsolutePath(), isSystem));
105       if (relativeArgs != null) {
106         final String relative = CUtil.getRelativePath(baseDirPath, includeDir);
107         relativeArgs.addElement(getIncludeDirSwitch(relative, isSystem));
108         if (includePathId != null) {
109           if (includePathId.length() == 0) {
110             includePathId.append("/I");
111           } else {
112             includePathId.append(" /I");
113           }
114           includePathId.append(relative);
115         }
116       }
117     }
118   }

122   protected void buildDefineArguments(final CompilerDef[] defs, final Vector<String> args) {
123     //
124     // assume that we aren't inheriting defines from containing <cc>
125     //
126     UndefineArgument[] merged = defs[0].getActiveDefines();
127     for (int i = 1; i < defs.length; i++) {
128       //
129       // if we are inheriting, merge the specific defines with the
130       // containing defines
131       merged = UndefineArgument.merge(defs[i].getActiveDefines(), merged);
132     }
133     final StringBuffer buf = new StringBuffer(30);
134     for (final UndefineArgument current : merged) {
135       buf.setLength(0);
136       if (current.isDefine()) {
137         getDefineSwitch(buf, current.getName(), current.getValue());
138       } else {
139         getUndefineSwitch(buf, current.getName());
140       }
141       args.addElement(buf.toString());
142     }
143   }

168   public void compile(final CCTask task, final File outputDir, final String[] sourceFiles, String[] args,
169       final String[] endArgs, final boolean relentless, final CommandLineCompilerConfiguration config,
170       final ProgressMonitor monitor) throws BuildException {
171     BuildException exc = null;
172     //
173     // determine length of executable name and args
174     //
175     String command = getCommandWithPath(config);
176     if (config.isUseCcache()) {
177       // Replace the command with "ccache" and push the old compiler
178       // command into the args.
179       final String compilerCommand = command;
180       command = CCACHE_CMD;
181       args = ObjectArrays.concat(compilerCommand, args);
182     }
183     int baseLength = command.length() + args.length + endArgs.length;
184     if (this.libtool) {
185       baseLength += 8;
186     }
187     for (final String arg : args) {
188       baseLength += arg.length();
189     }
190     for (final String endArg : endArgs) {
191       baseLength += endArg.length();
192     }
193     if (baseLength > getMaximumCommandLength()) {
194       throw new BuildException("Command line is over maximum length without specifying source file");
195     }
196     //
197     // typically either 1 or Integer.MAX_VALUE
198     //
199     final int maxInputFilesPerCommand = getMaximumInputFilesPerCommand();
200     final int argumentCountPerInputFile = getArgumentCountPerInputFile();
201     for (int sourceIndex = 0; sourceIndex < sourceFiles.length;) {
202       int cmdLength = baseLength;
203       int firstFileNextExec;
204       for (firstFileNextExec = sourceIndex; firstFileNextExec < sourceFiles.length
205           && firstFileNextExec - sourceIndex < maxInputFilesPerCommand; firstFileNextExec++) {
206         cmdLength += getTotalArgumentLengthForInputFile(outputDir, sourceFiles[firstFileNextExec]);
207         if (cmdLength >= getMaximumCommandLength()) {
208           break;
209         }
210       }
211       if (firstFileNextExec == sourceIndex) {
212         throw new BuildException("Extremely long file name, can't fit on command line");
213       }
214 
215       ArrayList<String> commandlinePrefix = new ArrayList<String>();
216       if (this.libtool) {
217         commandlinePrefix.add("libtool");
218       }
219       commandlinePrefix.add(command);
220       for (final String arg : args) {
221         commandlinePrefix.add(arg);
222       }
223 
224       int retval = 0;
225       for (int j = sourceIndex; j < firstFileNextExec; j++) {
226         ArrayList<String> commandlineSuffix = new ArrayList<String>();
227 
228         for (int k = 0; k < argumentCountPerInputFile; k++) {
229           commandlineSuffix.add(getInputFileArgument(outputDir, sourceFiles[j], k));
230         }
231         for (final String endArg : endArgs) {
232           commandlineSuffix.add(endArg);
233         }
234 
235         ArrayList<String> commandline = new ArrayList<String>(commandlinePrefix);
236         commandline.addAll(commandlineSuffix);
237         final int ret = runCommand(task, workDir,
238             commandline.toArray(new String[commandline.size()]));
239         if (ret != 0) { retval = ret; }
240       }
241       if (monitor != null) {
242         final String[] fileNames = new String[firstFileNextExec - sourceIndex];
243 
244         for (int j = 0; j < fileNames.length; j++) {
245           fileNames[j] = sourceFiles[sourceIndex + j];
246         }
247         monitor.progress(fileNames);
248       }
249       //
250       // if the process returned a failure code and
251       // we aren't holding an exception from an earlier
252       // interation
253       if (retval != 0 && exc == null) {
254         //
255         // construct the exception
256         //
257         exc = new BuildException(getCommandWithPath(config) + " failed with return code " + retval, task.getLocation());
258 
259         //
260         // and throw it now unless we are relentless
261         //
262         if (!relentless) {
263           throw exc;
264         }
265       }
266       sourceIndex = firstFileNextExec;
267     }
268     //
269     // if the compiler returned a failure value earlier
270     // then throw an exception
271     if (exc != null) {
272       throw exc;
273     }
274   }

276   @Override
277   protected CompilerConfiguration createConfiguration(final CCTask task, final LinkType linkType,
278       final ProcessorDef[] baseDefs, final CompilerDef specificDef, final TargetDef targetPlatform,
279       final VersionInfo versionInfo) {
280 
281     this.prefix = specificDef.getCompilerPrefix();
282     this.objDir = task.getObjdir();
283     final Vector<String> args = new Vector<String>();
284     final CompilerDef[] defaultProviders = new CompilerDef[baseDefs.length + 1];
285     for (int i = 0; i < baseDefs.length; i++) {
286       defaultProviders[i + 1] = (CompilerDef) baseDefs[i];
287     }
288     defaultProviders[0] = specificDef;
289     final Vector<CommandLineArgument> cmdArgs = new Vector<CommandLineArgument>();
290 
291     //
292     // add command line arguments inherited from <cc> element
293     // any "extends" and finally the specific CompilerDef
294     CommandLineArgument[] commandArgs;
295     for (int i = defaultProviders.length - 1; i >= 0; i--) {
296       commandArgs = defaultProviders[i].getActiveProcessorArgs();
297       for (final CommandLineArgument commandArg : commandArgs) {
298         if (commandArg.getLocation() == 0) {
299           String arg = commandArg.getValue();
300           if (isWindows() && arg.matches(".*[ \"].*")) {
301             // Work around inconsistent quoting by Ant
302             arg = "\"" + arg.replaceAll("[\\\\\"]", "\\\\$0") + "\"";
303           }
304           args.addElement(arg);
305         } else {
306           cmdArgs.addElement(commandArg);
307         }
308       }
309     }
310     final Vector<ProcessorParam> params = new Vector<ProcessorParam>();
311     //
312     // add command line arguments inherited from <cc> element
313     // any "extends" and finally the specific CompilerDef
314     ProcessorParam[] paramArray;
315     for (int i = defaultProviders.length - 1; i >= 0; i--) {
316       paramArray = defaultProviders[i].getActiveProcessorParams();
317       for (final ProcessorParam element : paramArray) {
318         params.add(element);
319       }
320     }
321     paramArray = params.toArray(new ProcessorParam[params.size()]);
322 
323     if (specificDef.isClearDefaultOptions() == false) {
324       final boolean multithreaded = specificDef.getMultithreaded(defaultProviders, 1);
325       final boolean debug = specificDef.getDebug(baseDefs, 0);
326       final boolean exceptions = specificDef.getExceptions(defaultProviders, 1);
327       final Boolean rtti = specificDef.getRtti(defaultProviders, 1);
328       final OptimizationEnum optimization = specificDef.getOptimization(defaultProviders, 1);
329       this.addImpliedArgs(args, debug, multithreaded, exceptions, linkType, rtti, optimization);
330     }
331 
332     //
333     // add all appropriate defines and undefines
334     //
335     buildDefineArguments(defaultProviders, args);
336     final int warnings = specificDef.getWarnings(defaultProviders, 0);
337     addWarningSwitch(args, warnings);
338     Enumeration<CommandLineArgument> argEnum = cmdArgs.elements();
339     int endCount = 0;
340     while (argEnum.hasMoreElements()) {
341       final CommandLineArgument arg = argEnum.nextElement();
342       switch (arg.getLocation()) {
343         case 1:
344           args.addElement(arg.getValue());
345           break;
346         case 2:
347           endCount++;
348           break;
349       }
350     }
351     final String[] endArgs = new String[endCount];
352     argEnum = cmdArgs.elements();
353     int index = 0;
354     while (argEnum.hasMoreElements()) {
355       final CommandLineArgument arg = argEnum.nextElement();
356       if (arg.getLocation() == 2) {
357         endArgs[index++] = arg.getValue();
358       }
359     }
360     //
361     // Want to have distinct set of arguments with relative
362     // path names for includes that are used to build
363     // the configuration identifier
364     //
365     final Vector<String> relativeArgs = (Vector) args.clone();
366     //
367     // add all active include and sysincludes
368     //
369     final StringBuffer includePathIdentifier = new StringBuffer();
370     final File baseDir = specificDef.getProject().getBaseDir();
371     String baseDirPath;
372     try {
373       baseDirPath = baseDir.getCanonicalPath();
374     } catch (final IOException ex) {
375       baseDirPath = baseDir.toString();
376     }
377     final Vector<String> includePath = new Vector<String>();
378     final Vector<String> sysIncludePath = new Vector<String>();
379     for (int i = defaultProviders.length - 1; i >= 0; i--) {
380       String[] incPath = defaultProviders[i].getActiveIncludePaths();
381       for (final String element : incPath) {
382         includePath.addElement(element);
383       }
384       incPath = defaultProviders[i].getActiveSysIncludePaths();
385       for (final String element : incPath) {
386         sysIncludePath.addElement(element);
387       }
388     }
389     final File[] incPath = new File[includePath.size()];
390     for (int i = 0; i < includePath.size(); i++) {
391       incPath[i] = new File(includePath.elementAt(i));
392     }
393     final File[] sysIncPath = new File[sysIncludePath.size()];
394     for (int i = 0; i < sysIncludePath.size(); i++) {
395       sysIncPath[i] = new File(sysIncludePath.elementAt(i));
396     }
397     addIncludes(baseDirPath, incPath, args, relativeArgs, includePathIdentifier, false);
398     addIncludes(baseDirPath, sysIncPath, args, null, null, true);
399     final StringBuffer buf = new StringBuffer(getIdentifier());
400     for (int i = 0; i < relativeArgs.size(); i++) {
401       buf.append(' ');
402       buf.append(relativeArgs.elementAt(i));
403     }
404     for (final String endArg : endArgs) {
405       buf.append(' ');
406       buf.append(endArg);
407     }
408     final String configId = buf.toString();
409     final String[] argArray = new String[args.size()];
410     args.copyInto(argArray);
411     final boolean rebuild = specificDef.getRebuild(baseDefs, 0);
412     final File[] envIncludePath = getEnvironmentIncludePath();
413     final String path = specificDef.getToolPath();
414 
415     CommandLineCompiler compiler = this;
416     Environment environment = specificDef.getEnv();
417     if (environment == null) {
418       for (final ProcessorDef baseDef : baseDefs) {
419         environment = baseDef.getEnv();
420         if (environment != null) {
421           compiler = (CommandLineCompiler) compiler.changeEnvironment(baseDef.isNewEnvironment(), environment);
422         }
423       }
424     } else {
425       compiler = (CommandLineCompiler) compiler.changeEnvironment(specificDef.isNewEnvironment(), environment);
426     }
427     return new CommandLineCompilerConfiguration(compiler, configId, incPath, sysIncPath, envIncludePath,
428         includePathIdentifier.toString(), argArray, paramArray, rebuild, endArgs, path, specificDef.getCcache());
429   }

495   protected String getInputFileArgument(final File outputDir, final String filename, final int index) {
496     //
497     // if there is an embedded space,
498     // must enclose in quotes
499     String relative="";
500     String inputFile;
501     try {
502       relative = FileUtils.getRelativePath(workDir, new File(filename));
503     } catch (Exception ex) {
504     }
505     if (relative.isEmpty()) {
506       inputFile = filename;
507     } else {
508       inputFile = relative;
509     }
510     if (inputFile.indexOf(' ') >= 0) {
511       final StringBuffer buf = new StringBuffer("\"");
512       buf.append(inputFile);
513       buf.append("\"");
514       return buf.toString();
515     }
516     return inputFile;
517   }
