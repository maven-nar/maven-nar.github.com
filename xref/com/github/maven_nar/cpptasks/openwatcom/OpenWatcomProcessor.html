30    /**
31     * Adds warning command line options.
32     *
33     * @param args Vector list of options
34     * @param level int value of WarningLevelEnum
35     */
36    public static void addWarningSwitch(final Vector args, final int level) {
37      switch (level) {
38        case 0:
39          args.addElement("/w0");
40          break;
41        case 1:
42          args.addElement("/w1");
43          break;
44        case 2:
45          break;
46        case 3:
47          args.addElement("/w2");
48          break;
49        case 4:
50          args.addElement("/w3");
51          break;
52        case 5:
53          args.addElement("/we");
54          break;
55        default:
56          args.addElement("/w1");
57          break;
58      }
59    }

61    /**
62     * Gets command line option to read from an option file.
63     *
64     * @param cmdFile String file name for option file
65     * @return String Command line option
66     */
67    public static String getCommandFileSwitch(final String cmdFile) {
68      StringBuffer buf = new StringBuffer("@");
69      if (cmdFile.indexOf(' ') >= 0) {
70        buf.append('\"');
71        buf.append(cmdFile.replace('/', '\\'));
72        buf.append('\"');
73      } else {
74        buf.append(cmdFile);
75      }
76      return buf.toString();
77    }
78  
79    /**
80     * Creates a command line option to define a preprocessor macro.
81     *
82     * @param buffer StringBuffer destination buffer
83     * @param define String parameter to define
84     * @param value String value, may be null
85     */
86    public static void getDefineSwitch(final StringBuffer buffer,
87                                       final String define,
88                                       final String value) {
89      buffer.append("/d");
90      buffer.append(define);
91      if (value != null && value.length() > 0) {
92        buffer.append('=');
93        buffer.append(value);
94      }
95    }
96  
97    /**
98     * Create a command line option to add a directory to the include path.
99     * @param includeDir String directory
100    * @return String command line option
101    */
102   public static String getIncludeDirSwitch(final String includeDir) {
103     return "/i=" + includeDir.replace('/', '\\');
104   }

106   /**
107    * Builds command line options to specify the output file names.
108    *
109    * @param outPath String path to output file
110    * @return String[] command line options
111    */
112   public static String[] getOutputFileSwitch(final String outPath) {
113     StringBuffer buf = new StringBuffer("/fo=");
114     if (outPath.indexOf(' ') >= 0) {
115       buf.append('\"');
116       buf.append(outPath);
117       buf.append('\"');
118     } else {
119       buf.append(outPath);
120     }
121     String[] retval = new String[] {
122         buf.toString()};
123     return retval;
124   }
125 
126   /**
127    * Get file selectors for specified libraries.
128    * @param libnames library names
129    * @param libType library type
130    * @return file selectors
131    */
132   public static String[] getLibraryPatterns(final String[] libnames,
133                                             final LibraryTypeEnum libType) {
134     StringBuffer buf = new StringBuffer();
135     String[] patterns = new String[libnames.length];
136     for (int i = 0; i < libnames.length; i++) {
137       buf.setLength(0);
138       buf.append(libnames[i]);
139       buf.append(".lib");
140       patterns[i] = buf.toString();
141     }
142     return patterns;
143   }

145   /**
146    * Builds a command line option to undefine a preprocessor macro.
147    * @param buffer StringBuffer destination

149    */
150   public static void getUndefineSwitch(final StringBuffer buffer,
151                                        final String define) {
152     buffer.append("/u");
153     buffer.append(define);
154   }
