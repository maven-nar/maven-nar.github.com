76      private static class SystemLibraryCollector implements FileVisitor {
77          private Hashtable<String, File> libraries;
78          private Linker linker;
79          public SystemLibraryCollector(Linker linker, Hashtable<String, File> libraries) {
80              this.linker = linker;
81              this.libraries = libraries;
82          }
83          public void visit(File basedir, String filename) {
84              if (linker.bid(filename) > 0) {
85                  File libfile = new File(basedir, filename);
86                  String key = linker.getLibraryKey(libfile);
87                  libraries.put(key, libfile);
88              }
89          }
90      }

113      * Builds a Hashtable to targets needing to be rebuilt keyed by compiler
114      * configuration
115      */
116     public static Map<ProcessorConfiguration, Vector<TargetInfo>> getTargetsToBuildByConfiguration(Map<String, TargetInfo> targets) {
117         Map<ProcessorConfiguration, Vector<TargetInfo>> targetsByConfig = new HashMap<ProcessorConfiguration, Vector<TargetInfo>>();
118         Iterator<TargetInfo> targetEnum = targets.values().iterator();
119         while (targetEnum.hasNext()) {
120             TargetInfo target = targetEnum.next();
121             if (target.getRebuild()) {
122                 Vector<TargetInfo> targetsForSameConfig = targetsByConfig
123                         .get(target.getConfiguration());
124                 if (targetsForSameConfig != null) {
125                     targetsForSameConfig.addElement(target);
126                 } else {
127                     targetsForSameConfig = new Vector<TargetInfo>();
128                     targetsForSameConfig.addElement(target);
129                     targetsByConfig.put(target.getConfiguration(),
130                             targetsForSameConfig);
131                 }
132             }
133         }
134         return targetsByConfig;
135     }

350     /**
351      * Checks all targets that are not forced to be rebuilt or are missing
352      * object files to be checked for modified include files
353      * 
354      * @return total number of targets to be rebuilt
355      *  
356      */
357     protected int checkForChangedIncludeFiles(Map<String, TargetInfo> targets) {
358         int potentialTargets = 0;
359         int definiteTargets = 0;
360         Iterator<TargetInfo> targetEnum = targets.values().iterator();
361         while (targetEnum.hasNext()) {
362             TargetInfo target = targetEnum.next();
363             if (!target.getRebuild()) {
364                 potentialTargets++;
365             } else {
366                 definiteTargets++;
367             }
368         }
369         //
370         //     If there were remaining targets that
371         //        might be out of date
372         //
373         if (potentialTargets > 0) {
374             log("Starting dependency analysis for "
375                     + Integer.toString(potentialTargets) + " files.");
376             DependencyTable dependencyTable = new DependencyTable(_objDir);
377             try {
378                 dependencyTable.load();
379             } catch (Exception ex) {
380                 log("Problem reading dependencies.xml: " + ex.toString());
381             }
382             targetEnum = targets.values().iterator();
383             while (targetEnum.hasNext()) {
384                 TargetInfo target = targetEnum.next();
385                 if (!target.getRebuild()) {
386                     if (dependencyTable.needsRebuild(this, target,
387                             dependencyDepth)) {
388                         target.mustRebuild();
389                     }
390                 }
391             }
392             dependencyTable.commit(this);
393         }
394         //
395         //   count files being rebuilt now
396         //
397         int currentTargets = 0;
398         targetEnum = targets.values().iterator();
399         while (targetEnum.hasNext()) {
400             TargetInfo target = targetEnum.next();
401             if (target.getRebuild()) {
402                 currentTargets++;
403             }
404         }
405         if (potentialTargets > 0) {
406             log(Integer.toString(potentialTargets - currentTargets
407                     + definiteTargets)
408                     + " files are up to date.");
409             log(Integer.toString(currentTargets - definiteTargets)
410                     + " files to be recompiled from dependency analysis.");
411         }
412         log(Integer.toString(currentTargets) + " total files to be compiled.");
413         return currentTargets;
414     }
415     protected LinkerConfiguration collectExplicitObjectFiles(
416             Vector<File> objectFiles, Vector<File> sysObjectFiles, VersionInfo versionInfo) {
417         //
418         //    find the first eligible linker
419         //
420         //
421         ProcessorConfiguration linkerConfig = null;
422         LinkerDef selectedLinkerDef = null;
423         Linker selectedLinker = null;
424         Hashtable<String, File> sysLibraries = new Hashtable<String, File>();
425         TargetDef targetPlatform = getTargetPlatform();
426         FileVisitor objCollector = null;
427         FileVisitor sysLibraryCollector = null;
428         for (int i = 0; i < _linkers.size(); i++) {
429             LinkerDef currentLinkerDef = (LinkerDef) _linkers.elementAt(i);
430             if (currentLinkerDef.isActive()) {
431                 selectedLinkerDef = currentLinkerDef;
432                 selectedLinker = currentLinkerDef.getProcessor().getLinker(
433                         linkType);
434                 //
435                 //   skip the linker if it doesn't know how to
436                 //      produce the specified link type
437                 if (selectedLinker != null) {
438                     linkerConfig = currentLinkerDef.createConfiguration(this,
439                             linkType, linkerDef, targetPlatform, versionInfo);
440                     if (linkerConfig != null) {
441                         //
442                         //   create collectors for object files
443                         //      and system libraries
444                         objCollector = new ObjectFileCollector(selectedLinker,
445                                 objectFiles);
446                         sysLibraryCollector = new SystemLibraryCollector(
447                                 selectedLinker, sysLibraries);
448                         //
449                         //    if the <linker> has embedded <fileset>'s
450                         //       (such as linker specific libraries)
451                         //       add them as object files.
452                         //
453                         if (currentLinkerDef.hasFileSets()) {
454                             currentLinkerDef.visitFiles(objCollector);
455                         }
456                         //
457                         //    user libraries are just a specialized form
458                         //       of an object fileset
459                         selectedLinkerDef.visitUserLibraries(selectedLinker,
460                                 objCollector);
461                     }
462                     break;
463                 }
464             }
465         }
466         if (linkerConfig == null) {
467             linkerConfig = linkerDef.createConfiguration(this, linkType, null, targetPlatform, versionInfo);
468             selectedLinker = (Linker) linkerDef.getProcessor().getLinker(
469                     linkType);
470             objCollector = new ObjectFileCollector(selectedLinker, objectFiles);
471             sysLibraryCollector = new SystemLibraryCollector(selectedLinker,
472                     sysLibraries);
473         }
474         //
475         //  unless there was a <linker> element that
476         //     explicitly did not inherit files from
477         //        containing <cc> element
478         if (selectedLinkerDef == null || selectedLinkerDef.getInherit()) {
479             linkerDef.visitUserLibraries(selectedLinker, objCollector);
480             linkerDef.visitSystemLibraries(selectedLinker, sysLibraryCollector);
481         }
482         //
483         //   if there was a <syslibset> in a nested <linker>
484         //       evaluate it last so it takes priority over
485         //       identically named libs from <cc> element
486         //
487         if (selectedLinkerDef != null) {
488             //
489             //    add any system libraries to the hashtable
490             //       done in reverse order so the earliest
491             //        on the classpath takes priority
492             selectedLinkerDef.visitSystemLibraries(selectedLinker,
493                     sysLibraryCollector);
494         }
495         //
496         //   copy over any system libraries to the
497         //      object files vector
498         //
499         Enumeration<File> sysLibEnum = sysLibraries.elements();
500         while (sysLibEnum.hasMoreElements()) {
501             sysObjectFiles.addElement(sysLibEnum.nextElement());
502         }
503         return (LinkerConfiguration) linkerConfig;
504     }

537     /**
538      * Executes the task. Compiles the given files.
539      * 
540      * @throws BuildException
541      *             if someting goes wrong with the build
542      */
543     public void execute() throws BuildException {
544         //
545         //   if link type allowed objdir to be defaulted
546         //      provide it from outfile
547         if (_objDir == null) {
548         	if(_outfile != null) {
549         		_objDir = new File(_outfile.getParent());
550         	} else {
551         		_objDir = new File(".");
552         	}
553         }
554         
555         //
556         //   if the object directory does not exist
557         //
558         if (!_objDir.exists()) {
559             throw new BuildException("Object directory does not exist");
560         }
561         TargetHistoryTable objHistory = new TargetHistoryTable(this, _objDir);
562 
563         //
564         //   get the first active version info
565         //
566         VersionInfo versionInfo = null;
567         Enumeration<VersionInfo> versionEnum = versionInfos.elements();
568         while (versionEnum.hasMoreElements()) {
569         	versionInfo = (VersionInfo) versionEnum.nextElement();
570         	versionInfo = versionInfo.merge();
571         	if (versionInfo.isActive()) {
572         		break;
573         	} else {
574         		versionInfo = null;
575         	}
576         }
577         
578         
579         //
580         //  determine the eventual linker configuration
581         //      (may be null) and collect any explicit
582         //          object files or libraries
583         Vector<File> objectFiles = new Vector<File>();
584         Vector<File> sysObjectFiles = new Vector<File>();
585         LinkerConfiguration linkerConfig = collectExplicitObjectFiles(
586                 objectFiles, sysObjectFiles, versionInfo);
587         
588         
589         //
590         //   Assemble hashtable of all files
591         //       that we know how to compile (keyed by output file name)
592         //
593         Map<String, TargetInfo> targets = getTargets(linkerConfig, objectFiles, versionInfo, _outfile);
594         TargetInfo linkTarget = null;
595         //
596         //   if output file is not specified,
597         //      then skip link step
598         //
599         if (_outfile != null) {
600             linkTarget = getLinkTarget(linkerConfig, objectFiles,
601                     sysObjectFiles, targets, versionInfo);
602         }
603         
604         if (projects.size() > 0) {
605         	List<File> files = new ArrayList<File>();
606         	ProjectFileCollector matcher = new ProjectFileCollector(files);
607             for (int i = 0; i < _compilers.size(); i++) {
608                 CompilerDef currentCompilerDef = (CompilerDef) _compilers
609                         .elementAt(i);
610                 if (currentCompilerDef.isActive()) {
611                     if (currentCompilerDef.hasFileSets()) {
612                         currentCompilerDef.visitFiles(matcher);
613                     }
614                 }
615             }
616             compilerDef.visitFiles(matcher);
617         	
618         	
619         	Enumeration<ProjectDef> iter = projects.elements();
620         	while (iter.hasMoreElements()) {
621         		ProjectDef projectDef = iter.nextElement();
622         		if (projectDef.isActive()) {
623         			projectDef.execute(this, files, targets, linkTarget);
624         		}
625         	}
626         }
627         if (projectsOnly) return;
628       
629         
630         
631         //
632         //     mark targets that don't have a history record or
633         //        whose source last modification time is not
634         //        the same as the history to be rebuilt
635         //
636         objHistory.markForRebuild(targets);
637         CCTaskProgressMonitor monitor = new CCTaskProgressMonitor(objHistory, versionInfo);
638         //
639         //      check for changed include files
640         //
641         int rebuildCount = checkForChangedIncludeFiles(targets);
642         if (rebuildCount > 0) {
643             BuildException compileException = null;
644             //
645             //    compile all targets with getRebuild() == true
646             //
647             Map<ProcessorConfiguration, Vector<TargetInfo>> targetsByConfig =
648                     getTargetsToBuildByConfiguration(targets);
649             //
650             //    build array containing Vectors with precompiled generation
651             //       steps going first
652             //
653             Vector<TargetInfo>[] targetVectors = new Vector[targetsByConfig.size()];
654             int index = 0;
655             Iterator<Vector<TargetInfo>> targetVectorEnum = targetsByConfig.values().iterator();
656             while (targetVectorEnum.hasNext()) {
657                 Vector<TargetInfo> targetsForConfig = targetVectorEnum.next();
658                 //
659                 //    get the configuration from the first entry
660                 //
661                 CompilerConfiguration config = (CompilerConfiguration) ((TargetInfo) targetsForConfig
662                         .elementAt(0)).getConfiguration();
663                 if (config.isPrecompileGeneration()) {
664                     targetVectors[index++] = targetsForConfig;
665                 }
666             }
667             targetVectorEnum = targetsByConfig.values().iterator();
668             while (targetVectorEnum.hasNext()) {
669                 Vector<TargetInfo> targetsForConfig = targetVectorEnum.next();
670                 for (int i = 0; i < targetVectors.length; i++) {
671                     if (targetVectors[i] == targetsForConfig) {
672                         break;
673                     }
674                     if (targetVectors[i] == null) {
675                         targetVectors[i] = targetsForConfig;
676                         break;
677                     }
678                 }
679             }
680             // BEGINFREEHEP
681             Progress progress = new Progress(getObjdir(), rebuildCount);
682             progress.start();
683             // ENDFREEHEP
684             
685             for (int i = 0; i < targetVectors.length; i++) {
686                 //
687                 //    get the targets for this configuration
688                 //
689                 Vector<TargetInfo> targetsForConfig = targetVectors[i];
690                 //
691                 //    get the configuration from the first entry
692                 //
693                 CompilerConfiguration config = (CompilerConfiguration) ((TargetInfo) targetsForConfig
694                         .elementAt(0)).getConfiguration();
695                 //
696                 //    prepare the list of source files
697                 //
698                 
699                 // BEGINFREEHEP
700                 int noOfCores = Runtime.getRuntime().availableProcessors();
701                 log("Found "+noOfCores+" processors available");
702                 if (maxCores > 0) {
703                     noOfCores = Math.min(maxCores, noOfCores);
704                     log("Limited processors to "+noOfCores);
705                 }
706                 int noOfFiles = targetsForConfig.size();
707                 if (noOfFiles < noOfCores) {
708                     noOfCores = noOfFiles;
709                     log("Limited used processors to "+noOfCores);                   
710                 }
711                 if (ordered) {
712                 	noOfCores = 1;
713                 	log("Limited processors to 1 due to ordering of source files");
714                 }
715                 
716                 List<String>[] sourceFiles = new List[noOfCores];
717                 for (int j = 0; j < sourceFiles.length; j++) {
718                     sourceFiles[j] = new ArrayList<String>(noOfFiles / sourceFiles.length);
719                 }
720                 Enumeration<TargetInfo> targetsEnum = targetsForConfig.elements();
721                 index = 0;
722                 while (targetsEnum.hasMoreElements()) {
723                     TargetInfo targetInfo = targetsEnum.nextElement();
724                     sourceFiles[index++].add(targetInfo.getSources()[0]
725                             .toString());
726                     index %= sourceFiles.length;
727                 }
728 
729                 // setup cores/cpus
730                 Core[] cores = new Core[noOfCores];
731                 for (int j = 0; j < cores.length; j++) {
732                     cores[j] = new Core(this, j, config, _objDir, sourceFiles[j],
733                             relentless, monitor);
734                     log("\nStarting Core " + j + " with "
735                             + sourceFiles[j].size() + " source files...");
736                 }
737 
738                 // starting cores
739                 for (int j = 0; j < cores.length; j++) {
740                     cores[j].start();
741                 }
742                 
743                 // checking cores
744                 boolean alive = false;
745                 try {
746                     do {
747                         alive = false;
748                         for (int j = 0; j < cores.length; j++) {
749                             if (cores[j] != null) {
750                                 if (cores[j].isAlive()) {
751                                     alive = true;
752                                 } else {
753                                     Exception exception = cores[j].getException();
754                                     if (exception != null) {
755                                         if ((compileException == null) && (exception instanceof BuildException)) {
756                                             compileException = (BuildException)exception;
757                                         } else {
758                                             log(cores[j].getName()+" "+exception+"                                  ", Project.MSG_ERR);
759                                         }
760                                         if (!relentless) {
761                                             cores[j] = null;
762                                             alive = false;
763                                             break;
764                                         }
765                                     }
766                                     cores[j] = null;
767                                 }
768                             }
769                         }
770                         if (alive) {
771                             // wait for a maximum of 5 seconds or #files*2 seconds.
772                             Thread.sleep(Math.min(5000, sourceFiles[0].size()*2000));
773                         }
774                     } while (alive);
775                 } catch (InterruptedException e) {
776                     break;
777                 }
778 
779                 // killing leftovers
780                 for (int j = 0; j < cores.length; j++) {
781                     if (cores[j] != null) {
782                         cores[j].interrupt();
783                         log(cores[j].getName()+" interrupted                                          ");
784                     }
785                 }
786                 
787                 if ((!relentless) && (compileException != null)) {
788                     break;
789                 }
790                 // ENDFREEHEP
791 
792                 
793                 /* OLD CODE
794                 String[] sourceFiles = new String[targetsForConfig.size()];
795                 Enumeration targetsEnum = targetsForConfig.elements();
796                 index = 0;
797                 while (targetsEnum.hasMoreElements()) {
798                     TargetInfo targetInfo = ((TargetInfo) targetsEnum
799                             .nextElement());
800                     sourceFiles[index++] = targetInfo.getSources()[0]
801                             .toString();
802                 }
803                 try {
804                     config.compile(this, _objDir, sourceFiles, relentless,
805                             monitor);
806                 } catch (BuildException ex) {
807                     if (compileException == null) {
808                         compileException = ex;
809                     }
810                     if (!relentless)
811                         break;
812                 }
813                 */
814             }
815             
816             // BEGINFREEHEP
817             progress.exit();
818             try {
819                 progress.join();
820             } catch (InterruptedException ex) {
821             }
822             // ENDFREEHEP
823 
824             //
825             //   save the details of the object file compilation
826             //     settings to disk for dependency analysis
827             //
828             try {
829                 objHistory.commit();
830             } catch (IOException ex) {
831                 this.log("Error writing history.xml: " + ex.toString());
832             }
833             //
834             //  if we threw a compile exception and
835             //     didn't throw it at the time because
836             //     we were relentless then
837             //        save the history and
838             //           throw the exception
839             //
840             if (compileException != null) {
841             	if (failOnError) {
842             		throw compileException;
843             	} else {
844                     log(compileException.getMessage(), Project.MSG_ERR);
845                     return;
846                 }
847             }
848         }
849         //
850         //    if the dependency tree was not fully
851         //      evaluated, then throw an exception
852         //      since we really didn't do what we
853         //      should have done
854         //
855         //
856         if (dependencyDepth >= 0) {
857             throw new BuildException(
858                     "All files at depth "
859                             + Integer.toString(dependencyDepth)
860                             + " from changes successfully compiled.\n"
861                             + "Remove or change dependencyDepth to -1 to perform full compilation.");
862         }
863         //
864         //   if no link target then
865         //       commit the history for the object files
866         //           and leave the task
867         if (linkTarget != null) {
868             //
869             //    get the history for the link target (may be the same
870             //        as the object history)
871             TargetHistoryTable linkHistory = getLinkHistory(objHistory);
872             //
873             //    see if it needs to be rebuilt
874             //
875             linkHistory.markForRebuild(linkTarget);
876             //
877             //    if it needs to be rebuilt, rebuild it
878             //
879             File output = linkTarget.getOutput();
880             if (linkTarget.getRebuild()) {
881                 LinkerConfiguration linkConfig = (LinkerConfiguration) linkTarget
882                         .getConfiguration();
883                 // BEGINFREEHEP
884                 log("Linking...");
885                 log("Starting link {" + linkConfig.getIdentifier() + "}");
886                 // ENDFREEHEP
887                 if (failOnError) {
888                 	linkConfig.link(this, linkTarget);
889                 } else {
890                 	try {
891                     	linkConfig.link(this, linkTarget);                		
892                 	} catch(BuildException ex) {
893                         log(ex.getMessage(), Project.MSG_ERR);
894                         return;                		
895                 	}
896                 }
897                 if (outputFileProperty != null)
898                     getProject().setProperty(outputFileProperty,
899                             output.getAbsolutePath());
900                 linkHistory.update(linkTarget);
901                 try {
902                     linkHistory.commit();
903                 } catch (IOException ex) {
904                     log("Error writing link history.xml: " + ex.toString());
905                 }
906             } else {
907                 if (outputFileProperty != null)
908                     getProject().setProperty(outputFileProperty,
909                             output.getAbsolutePath());
910             }
911         }
912     }

1069     /**
1070      * This method collects a Hashtable, keyed by output file name, of
1071      * TargetInfo's for every source file that is specified in the filesets of
1072      * the <cc>and nested <compiler>elements. The TargetInfo's contain the
1073      * appropriate compiler configurations for their possible compilation
1074      *  
1075      */
1076     private Map<String, TargetInfo> getTargets(LinkerConfiguration linkerConfig,
1077             Vector<File> objectFiles, VersionInfo versionInfo, File outputFile) {
1078         // FREEHEP
1079         final List<String> order = new ArrayList<String>();
1080         
1081         Map<String, TargetInfo> targets = new TreeMap<String, TargetInfo>(new Comparator<String>() {
1082             // Order according to "order" List followed by alphabetical order
1083             public int compare(String f0, String f1) {
1084                 if (order.isEmpty()) return f0.compareTo(f1);
1085 
1086                 // Not sure why DUNS is trimming the trailing file extension - but this breaks multi compilers like resource/midl if same basename as c/cpp
1087                 // so moved the comparison to above to avoid trimming the extension if there is no order, which half fixes the issue.
1088 
1089                 f0 = f0.lastIndexOf('.') < 0 ? f0 : f0.substring(0, f0.lastIndexOf('.'));
1090                 f1 = f1.lastIndexOf('.') < 0 ? f1 : f1.substring(0, f1.lastIndexOf('.'));
1091 
1092                 // make sure we use only one core
1093                 ordered = true;
1094                 
1095                 // order according to list or alphabetical
1096                 int i0 = order.indexOf(f0);
1097                 int i1 = order.indexOf(f1);
1098                 
1099                 if (i0 < 0) {
1100                 	if (i1 < 0) {
1101                 		// none in list
1102                 		return f0.compareTo(f1);
1103                 	} else {
1104                 		// i1 in list
1105                 		return +1;
1106                 	}
1107                 } else {
1108                 	if (i1 < 0) {
1109                 		// i0 in list
1110                 		return -1;
1111                 	} else {
1112                 		// both in list
1113                 		return i0 == i1 ? 0 : i0 < i1 ? -1 : +1;
1114                 	}
1115                 }
1116             }
1117         });
1118         
1119         TargetDef targetPlatform = getTargetPlatform();
1120         //
1121         //   find active (specialized) compilers
1122         //
1123         Vector<ProcessorConfiguration> biddingProcessors = new Vector<ProcessorConfiguration>(_compilers.size());
1124         for (int i = 0; i < _compilers.size(); i++) {
1125             CompilerDef currentCompilerDef = (CompilerDef) _compilers
1126                     .elementAt(i);
1127             if (currentCompilerDef.isActive()) {
1128                 ProcessorConfiguration config = currentCompilerDef
1129                         .createConfiguration(this, linkType, compilerDef, 
1130                         		targetPlatform, versionInfo);
1131                 //
1132                 //   see if this processor had a precompile child element
1133                 //
1134                 PrecompileDef precompileDef = currentCompilerDef
1135                         .getActivePrecompile(compilerDef);
1136                 ProcessorConfiguration[] localConfigs = new ProcessorConfiguration[]{config};
1137                 //
1138                 //    if it does then
1139                 //
1140                 if (precompileDef != null) {
1141                     File prototype = precompileDef.getPrototype();
1142                     //
1143                     //  will throw exceptions if prototype doesn't exist, etc
1144                     //
1145                     if (!prototype.exists()) {
1146                         throw new BuildException("prototype ("
1147                                 + prototype.toString() + ") does not exist.");
1148                     }
1149                     if (prototype.isDirectory()) {
1150                         throw new BuildException("prototype ("
1151                                 + prototype.toString() + ") is a directory.");
1152                     }
1153                     String[] exceptFiles = precompileDef.getExceptFiles();
1154                     //
1155                     //  create a precompile building and precompile using
1156                     //      variants of the configuration
1157                     //      or return null if compiler doesn't support
1158                     //      precompilation
1159                     CompilerConfiguration[] configs = ((CompilerConfiguration) config)
1160                             .createPrecompileConfigurations(prototype,
1161                                     exceptFiles);
1162                     if (configs != null && configs.length == 2) {
1163                         //
1164                         //   visit the precompiled file to add it into the
1165                         //      targets list (just like any other file if
1166                         //      compiler doesn't support precompilation)
1167                         TargetMatcher matcher = new TargetMatcher(this,
1168                                 _objDir,
1169                                 new ProcessorConfiguration[]{configs[0]},
1170                                 linkerConfig, objectFiles, targets, versionInfo);
1171      
1172                         matcher.visit(new File(prototype.getParent()),
1173                                 prototype.getName());
1174                         //
1175                         //   only the configuration that uses the
1176                         //      precompiled header gets added to the bidding list
1177                         biddingProcessors.addElement(configs[1]);
1178                         localConfigs = new ProcessorConfiguration[2];
1179                         localConfigs[0] = configs[1];
1180                         localConfigs[1] = config;
1181                     }
1182                 }
1183             	// BEGINFREEHEP
1184                 // a little trick here, the inner function needs the list to be final, so we repopulate it
1185                 order.clear();
1186             	List<String> newOrder = currentCompilerDef.getOrder();
1187             	if (newOrder != null) {
1188             		order.addAll(newOrder);
1189             	}
1190                 // ENDFREEHEP
1191             	
1192                 //
1193                 //   if the compiler has a fileset
1194                 //       then allow it to add its files
1195                 //       to the set of potential targets
1196                 if (currentCompilerDef.hasFileSets()) {
1197                     TargetMatcher matcher = new TargetMatcher(this, _objDir,
1198                             localConfigs, linkerConfig, objectFiles, targets,
1199 							versionInfo);
1200                     currentCompilerDef.visitFiles(matcher);
1201                 }
1202                 biddingProcessors.addElement(config);
1203             }
1204         }
1205         //
1206         //    add fallback compiler at the end
1207         //
1208         ProcessorConfiguration config = compilerDef.createConfiguration(this,
1209                 linkType, null, targetPlatform, versionInfo);
1210         biddingProcessors.addElement(config);
1211         ProcessorConfiguration[] bidders = new ProcessorConfiguration[biddingProcessors
1212                 .size()];
1213         biddingProcessors.copyInto(bidders);
1214         //
1215         //   bid out the <fileset>'s in the cctask
1216         //
1217         TargetMatcher matcher = new TargetMatcher(this, _objDir, bidders,
1218                 linkerConfig, objectFiles, targets, versionInfo);
1219         compilerDef.visitFiles(matcher);
1220         
1221         if (outputFile != null && versionInfo != null) {
1222         	boolean isDebug = linkerConfig.isDebug();
1223     		try {
1224     			linkerConfig.getLinker().addVersionFiles(versionInfo, linkType,
1225     				outputFile,
1226 					isDebug,
1227     				_objDir, matcher);
1228     		} catch(IOException ex) {
1229     			throw new BuildException(ex);
1230     		}
1231         }
1232         return targets;
1233     }
