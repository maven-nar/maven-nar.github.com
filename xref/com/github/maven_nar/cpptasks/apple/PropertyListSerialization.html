36  /**
37   * Static class that provides methods to serialize
38   * a Map to a Cocoa XML Property List.  Does not currently support
39   * date or data elements.
40   */
41  public final class PropertyListSerialization {
42      /**
43       * Private constructor.
44       */
45      private PropertyListSerialization() {
46  
47      }
48  
49      /**
50       * Serializes a property list into a Cocoa XML Property List document.
51       * @param propertyList property list.
52       * @param file destination.
53       * @param comments comments to insert into document.
54       * @throws SAXException if exception during serialization.
55       * @throws TransformerConfigurationException if exception creating serializer.
56       */
57      public static void serialize(final Map propertyList,
58                                   final List comments,
59                                   final File file)
60             throws IOException, SAXException,
61                 TransformerConfigurationException {
62          SAXTransformerFactory sf = (SAXTransformerFactory)
63                  SAXTransformerFactory.newInstance();
64          TransformerHandler handler = sf.newTransformerHandler();
65  
66          FileOutputStream os = new FileOutputStream(file);
67          StreamResult result = new StreamResult(os);
68          handler.setResult(result);
69  
70          handler.startDocument();
71          for(Iterator iter = comments.iterator(); iter.hasNext();) {
72              char[] comment = String.valueOf(iter.next()).toCharArray();
73              handler.comment(comment, 0, comment.length);
74          }
75          AttributesImpl attributes = new AttributesImpl();
76          handler.startElement(null, "plist", "plist", attributes);
77          serializeMap(propertyList, handler);
78          handler.endElement(null, "plist", "plist");
79  
80          handler.endDocument();
81      }

83      /**
84       * Serialize a map as a dict element.
85       * @param map map to serialize.
86       * @param handler destination of serialization events.
87       * @throws SAXException if exception during serialization.
88       */
89      private static void serializeMap(final Map map,
90                                       final ContentHandler handler)
91           throws SAXException {
92          AttributesImpl attributes = new AttributesImpl();
93          handler.startElement(null, "dict", "dict", attributes);
94  
95          if (map.size() > 0) {
96              //
97              //   need to output with sorted keys to maintain
98              //     reproducability
99              //
100             Object[] keys = map.keySet().toArray();
101             Arrays.sort(keys);
102             for(int i = 0; i < keys.length; i++) {
103                 String key = String.valueOf(keys[i]);
104                 handler.startElement(null, "key", "key", attributes);
105                 handler.characters(key.toCharArray(), 0, key.length());
106                 handler.endElement(null, "key", "key");
107                 serializeObject(map.get(keys[i]), handler);
108             }
109         }
110         handler.endElement(null, "dict", "dict");
111     }

113     /**
114      * Serialize a list as an array element.
115      * @param list list to serialize.
116      * @param handler destination of serialization events.
117      * @throws SAXException if exception during serialization.
118      */
119     private static void serializeList(final List list,
120                                       final ContentHandler handler)
121          throws SAXException {
122         AttributesImpl attributes = new AttributesImpl();
123         handler.startElement(null, "array", "array", attributes);
124         for(Iterator iter = list.iterator();iter.hasNext();) {
125             serializeObject(iter.next(), handler);
126         }
127         handler.endElement(null, "array", "array");
128     }

130     /**
131      * Creates an element with the specified tag name and character content.
132      * @param tag tag name.
133      * @param content character content.
134      * @param handler destination of serialization events.
135      * @throws SAXException if exception during serialization.
136      */
137     private static void serializeElement(final String tag,
138                                   final String content,
139                                   final ContentHandler handler)
140          throws SAXException {
141         AttributesImpl attributes = new AttributesImpl();
142         handler.startElement(null, tag, tag, attributes);
143         handler.characters(content.toCharArray(), 0, content.length());
144         handler.endElement(null, tag, tag);
145     }

152      * @throws SAXException if exception during serialization.

154     private static void serializeInteger(final Number integer,
155                                          final ContentHandler handler)
156          throws SAXException {
157         serializeElement("integer", String.valueOf(integer.longValue()),
158                 handler);
159     }

161     /**
162      * Serialize a Number as a real element.
163      * @param real number to serialize.
164      * @param handler destination of serialization events.
165      * @throws SAXException if exception during serialization.
166      */
167     private static void serializeReal(final Number real,
168                                       final ContentHandler handler)
169          throws SAXException {
170         serializeElement("real",
171                 String.valueOf(real.doubleValue()),
172                 handler);

176     /**
177      * Serialize a Boolean as a true or false element.
178      * @param val boolean to serialize.
179      * @param handler destination of serialization events.
180      * @throws SAXException if exception during serialization.
181      */
182     private static void serializeBoolean(final Boolean val,
183                                          final ContentHandler handler)
184          throws SAXException {
185         String tag = "false";
186         if (val.booleanValue()) {
187             tag = "true";
188         }
189         AttributesImpl attributes = new AttributesImpl();
190         handler.startElement(null, tag, tag, attributes);
191         handler.endElement(null, tag, tag);
192     }

194     /**
195      * Serialize a string as a string element.
196      * @param val string to serialize.
197      * @param handler destination of serialization events.
198      * @throws SAXException if exception during serialization.
199      */
200     private static void serializeString(final String val,
201                                         final ContentHandler handler)
202          throws SAXException {
203         serializeElement("string",
204                 val,
205                 handler);
206     }

209     /**
210      * Serialize an object using the best available element.
211      * @param obj object to serialize.
212      * @param handler destination of serialization events.
213      * @throws SAXException if exception during serialization.
214      */
215     private static void serializeObject(final Object obj,
216                                         final ContentHandler handler)
217         throws SAXException {
218         if (obj instanceof Map) {
219             serializeMap((Map) obj, handler);
220         } else if (obj instanceof List) {
221             serializeList((List) obj, handler);      
222         } else if (obj instanceof Number) {
223             if(obj instanceof Double
224                 ||obj instanceof Float) {
225                 serializeReal((Number) obj, handler);
226             } else {
227                 serializeInteger((Number) obj, handler);
228             }
229         } else if (obj instanceof Boolean) {
230             serializeBoolean((Boolean) obj, handler);
231         } else {
232             serializeString(String.valueOf(obj), handler);
233         }
234     }
