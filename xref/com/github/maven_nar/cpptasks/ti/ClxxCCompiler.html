33  /**
34   * Adapter for TI DSP compilers with cl** commands
35   * 
36   * @author CurtA
37   */
38  public class ClxxCCompiler extends CommandLineCCompiler {
39      /**
40       * Header file extensions
41       */
42      private static final String[] headerExtensions = new String[]{".h", ".hpp",
43              ".inl"};
44      /**
45       * Source file extensions
46       */
47      private static final String[] sourceExtensions = new String[]{".c", ".cc",
48              ".cpp", ".cxx", ".c++"};
49      /**
50       * Singleton for TMS320C55x
51       */
52      private static final ClxxCCompiler cl55 = new ClxxCCompiler("cl55", false,
53              null);
54      /**
55       * Singleton for TMS320C6000
56       */
57      private static final ClxxCCompiler cl6x = new ClxxCCompiler("cl6x", false,
58              null);
59      public static ClxxCCompiler getCl55Instance() {
60          return cl55;
61      }
62      public static ClxxCCompiler getCl6xInstance() {
63          return cl6x;
64      }
65      /**
66       * Private constructor
67       * 
68       * @param command
69       *            executable name
70       * @param newEnvironment
71       *            Change environment
72       * @param env
73       *            New environment
74       */
75      private ClxxCCompiler(String command, boolean newEnvironment,
76              Environment env) {
77          super(command, "-h", sourceExtensions, headerExtensions, ".o", false,
78                  null, newEnvironment, env);
79      }
80      protected void addImpliedArgs(
81              final Vector<String> args, 
82              final boolean debug,
83              final boolean multithreaded, 
84  			final boolean exceptions, 
85  			final LinkType linkType,
86  			final Boolean rtti,
87  			final OptimizationEnum optimization) {
88          if (debug) {
89              args.addElement("-gw");
90          }
91      }
92      protected void addWarningSwitch(Vector<String> args, int warnings) {
93          // TODO Auto-generated method stub
94      }
95      protected void getDefineSwitch(StringBuffer buffer, String define,
96              String value) {
97          buffer.append("-d");
98          buffer.append(define);
99          if (value != null) {
100             buffer.append('=');
101             buffer.append(value);
102         }
103     }
104     protected File[] getEnvironmentIncludePath() {
105         File[] c_dir = CUtil.getPathFromEnvironment("C_DIR", ";");
106         File[] cx_dir = CUtil.getPathFromEnvironment("C6X_C_DIR", ";");
107         if (c_dir.length == 0) {
108             return cx_dir;
109         }
110         if (cx_dir.length == 0) {
111             return c_dir;
112         }
113         File[] combo = new File[c_dir.length + cx_dir.length];
114         for (int i = 0; i < cx_dir.length; i++) {
115             combo[i] = cx_dir[i];
116         }
117         for (int i = 0; i < c_dir.length; i++) {
118             combo[i + cx_dir.length] = c_dir[i];
119         }
120         return combo;
121     }
122     protected String getIncludeDirSwitch(String source) {
123         return "-I" + source;
124     }
125     public Linker getLinker(LinkType type) {
126         if (type.isStaticLibrary()) {
127             if (this == cl6x) {
128                 return ClxxLibrarian.getCl6xInstance();
129             }
130             return ClxxLibrarian.getCl55Instance();
131         }
132         if (type.isSharedLibrary()) {
133             if (this == cl6x) {
134                 return ClxxLinker.getCl6xDllInstance();
135             }
136             return ClxxLinker.getCl55DllInstance();
137         }
138         if (this == cl6x) {
139             return ClxxLinker.getCl6xInstance();
140         }
141         return ClxxLinker.getCl55Instance();
142     }
143     public int getMaximumCommandLength() {
144         return 1024;
145     }
146     protected void getUndefineSwitch(StringBuffer buffer, String define) {
147         buffer.append("-u");
148         buffer.append(define);
149     }
