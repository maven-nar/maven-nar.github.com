63    private static String toProjectName(final String name) {
64        //
65        //    some characters are apparently not allowed in VS project names
66        //       but have not been able to find them documented
67        //       limiting characters to alphas, numerics and hyphens
68        StringBuffer projectNameBuf = new StringBuffer(name);
69        for (int i = 0; i < projectNameBuf.length(); i++) {
70          final char ch = projectNameBuf.charAt(i);
71          if (!((ch >= 'a' && ch <= 'z')
72                 || (ch >= 'A' && ch <= 'Z')
73                 || (ch >= '0' && ch <= '9'))) {
74            projectNameBuf.setCharAt(i, '_');
75          }
76        }
77        return projectNameBuf.toString();
78  
79    }

81    /**
82     *  Writes a project definition file.
83     * @param fileName File name base, writer may append appropriate extension
84     * @param task cc task for which to write project
85     * @param projectDef project element
86     * @param files source files
87     * @param targets compilation targets
88     * @param linkTarget link target
89     * @throws IOException if error writing project file
90     */
91    public void writeProject(final File fileName,
92                             final CCTask task,
93                             final ProjectDef projectDef,
94                             final List files,
95                             final Map targets,
96                             final TargetInfo linkTarget) throws IOException {
97  
98      //
99      //    some characters are apparently not allowed in VS project names
100     //       but have not been able to find them documented
101     //       limiting characters to alphas, numerics and hyphens
102     String projectName = projectDef.getName();
103     if (projectName != null) {
104       projectName = toProjectName(projectName);
105     } else {
106       projectName = toProjectName(fileName.getName());
107     }
108 
109     final String basePath = fileName.getAbsoluteFile().getParent();
110 
111     File dspFile = new File(fileName + ".dsp");
112     if (!projectDef.getOverwrite() && dspFile.exists()) {
113       throw new BuildException("Not allowed to overwrite project file "
114                                + dspFile.toString());
115     }
116     File dswFile = new File(fileName + ".dsw");
117     if (!projectDef.getOverwrite() && dswFile.exists()) {
118         throw new BuildException("Not allowed to overwrite project file "
119                                  + dswFile.toString());
120       }
121 
122     CommandLineCompilerConfiguration compilerConfig =
123         getBaseCompilerConfiguration(targets);
124     if (compilerConfig == null) {
125       throw new BuildException(
126           "Unable to generate Visual Studio project "
127           + "when Microsoft C++ is not used.");
128     }
129 
130     Writer writer = new BufferedWriter(new FileWriter(dspFile));
131     writer.write("# Microsoft Developer Studio Project File - Name=\"");
132     writer.write(projectName);
133     writer.write("\" - Package Owner=<4>\r\n");
134     writer.write(
135         "# Microsoft Developer Studio Generated Build File, Format Version ");
136     writer.write(this.version);
137     writer.write("\r\n");
138     writer.write("# ** DO NOT EDIT **\r\n\r\n");
139 
140 	writeComments(writer, projectDef.getComments());
141 
142     String outputType = task.getOuttype();
143     String subsystem = task.getSubsystem();
144     String targtype = "Win32 (x86) Dynamic-Link Library";
145     String targid = "0x0102";
146     if ("executable".equals(outputType)) {
147       if ("console".equals(subsystem)) {
148         targtype = "Win32 (x86) Console Application";
149         targid = "0x0103";
150       } else {
151         targtype = "Win32 (x86) Application";
152         targid = "0x0101";
153       }
154     } else if ("static".equals(outputType)) {
155       targtype = "Win32 (x86) Static Library";
156       targid = "0x0104";
157     }
158     writer.write("# TARGTYPE \"");
159     writer.write(targtype);
160     writer.write("\" ");
161     writer.write(targid);
162     writer.write("\r\n\r\nCFG=");
163 
164     writer.write(projectName + " - Win32 Debug");
165     writer.write("\r\n");
166 
167     writeMessage(writer, projectName, targtype);
168 
169     writer.write("# Begin Project\r\n");
170     if (version.equals("6.00")) {
171       writer.write("# PROP AllowPerConfigDependencies 0\r\n");
172     }
173     writer.write("# PROP Scc_ProjName \"\"\r\n");
174     writer.write("# PROP Scc_LocalPath \"\"\r\n");
175     writer.write("CPP=cl.exe\r\n");
176     writer.write("MTL=midl.exe\r\n");
177     writer.write("RSC=rc.exe\r\n");
178 
179     writer.write("\r\n!IF  \"$(CFG)\" == \"" + projectName + " - Win32 Release\"\r\n");
180 
181     writeConfig(writer, false, projectDef.getDependencies(), basePath, compilerConfig, linkTarget, targets);
182 
183     writer.write("\r\n!ELSEIF  \"$(CFG)\" == \"" + projectName + " - Win32 Debug\"\r\n");
184 
185     writeConfig(writer, true, projectDef.getDependencies(), basePath, compilerConfig, linkTarget, targets);
186 
187     writer.write("\r\n!ENDIF\r\n");
188 
189     writer.write("# Begin Target\r\n\r\n");
190     writer.write("# Name \"" + projectName + " - Win32 Release\"\r\n");
191     writer.write("# Name \"" + projectName + " - Win32 Debug\"\r\n");
192       
193 
194 
195     File[] sortedSources = getSources(files);
196 
197     if (version.equals("6.00")) {
198       final String sourceFilter = "cpp;c;cxx;rc;def;r;odl;idl;hpj;bat";
199       final String headerFilter = "h;hpp;hxx;hm;inl";
200       final String resourceFilter =
201           "ico;cur;bmp;dlg;rc2;rct;bin;rgs;gif;jpg;jpeg;jpe";
202 
203       writer.write("# Begin Group \"Source Files\"\r\n\r\n");
204       writer.write("# PROP Default_Filter \"" + sourceFilter + "\"\r\n");
205 
206       for (int i = 0; i < sortedSources.length; i++) {
207         if (!isGroupMember(headerFilter, sortedSources[i])
208             && !isGroupMember(resourceFilter, sortedSources[i])) {
209           writeSource(writer, basePath, sortedSources[i]);
210         }
211       }
212       writer.write("# End Group\r\n");
213 
214       writer.write("# Begin Group \"Header Files\"\r\n\r\n");
215       writer.write("# PROP Default_Filter \"" + headerFilter + "\"\r\n");
216 
217       for (int i = 0; i < sortedSources.length; i++) {
218         if (isGroupMember(headerFilter, sortedSources[i])) {
219           writeSource(writer, basePath, sortedSources[i]);
220         }
221       }
222       writer.write("# End Group\r\n");
223 
224       writer.write("# Begin Group \"Resource Files\"\r\n\r\n");
225       writer.write("# PROP Default_Filter \"" + resourceFilter + "\"\r\n");
226 
227       for (int i = 0; i < sortedSources.length; i++) {
228         if (isGroupMember(resourceFilter, sortedSources[i])) {
229           writeSource(writer, basePath, sortedSources[i]);
230         }
231       }
232       writer.write("# End Group\r\n");
233 
234     } else {
235       for (int i = 0; i < sortedSources.length; i++) {
236         writeSource(writer, basePath, sortedSources[i]);
237       }
238     }
239 
240     writer.write("# End Target\r\n");
241     writer.write("# End Project\r\n");
242     writer.close();
243 
244     //
245     //    write workspace file
246     //
247     writer = new BufferedWriter(new FileWriter(dswFile));
248     writeWorkspace(writer, projectDef, projectName, dspFile);
249     writer.close();
250 
251   }

253   private void writeConfig(final Writer writer,
254                            boolean isDebug,
255                            final List dependencies,
256                            final String basePath,
257                            CommandLineCompilerConfiguration compilerConfig,
258                            TargetInfo linkTarget,
259                            Map targets) throws IOException {
260       writer.write("# PROP BASE Use_MFC 0\r\n");
261 
262       String configType = "Release";
263       String configInt = "0";
264       String configMacro = "NDEBUG";
265       if (isDebug) {
266         configType = "Debug";
267         configInt = "1";
268         configMacro = "_DEBUG";
269       }
270 
271       writer.write("# PROP BASE Use_Debug_Libraries ");
272       writer.write(configInt);
273       writer.write("\r\n# PROP BASE Output_Dir \"");
274       writer.write(configType);
275       writer.write("\"\r\n");
276       writer.write("# PROP BASE Intermediate_Dir \"");
277       writer.write(configType);
278       writer.write("\"\r\n");
279       writer.write("# PROP BASE Target_Dir \"\"\r\n");
280       writer.write("# PROP Use_MFC 0\r\n");
281       writer.write("# PROP Use_Debug_Libraries ");
282       writer.write(configInt);
283       writer.write("\r\n# PROP Output_Dir \"");
284       writer.write(configType);
285       writer.write("\"\r\n");
286       writer.write("# PROP Intermediate_Dir \"");
287       writer.write(configType);
288       writer.write("\"\r\n");
289       writer.write("# PROP Target_Dir \"\"\r\n");
290       writeCompileOptions(writer, isDebug, basePath, compilerConfig);
291       writer.write(
292           "# ADD BASE MTL /nologo /D \"" + configMacro + "\" /mktyplib203 /o NUL /win32\r\n");
293       writer.write(
294           "# ADD MTL /nologo /D \"" + configMacro + "\" /mktyplib203 /o NUL /win32\r\n");
295       writer.write("# ADD BASE RSC /l 0x409 /d \"" + configMacro + "\"\r\n");
296       writer.write("# ADD RSC /l 0x409 /d \"" + configMacro + "\"\r\n");
297       writer.write("BSC32=bscmake.exe\r\n");
298       writer.write("# ADD BASE BSC32 /nologo\r\n");
299       writer.write("# ADD BSC32 /nologo\r\n");
300       writer.write("LINK32=link.exe\r\n");
301       writeLinkOptions(writer, isDebug, dependencies, basePath, linkTarget, targets);
302   }
303   private static void writeWorkspaceProject(final Writer writer,
304                                      final String projectName,
305                                      final String projectFile,
306                                      final List dependsOn) throws IOException {
307       writer.write("############################################");
308       writer.write("###################################\r\n\r\n");
309       String file = projectFile;
310       if(!file.startsWith(".") && !file.startsWith("\\") && !file.startsWith("/")) {
311           file = ".\\" + file;
312       }
313       writer.write("Project: \"" + projectName + "\"=\""
314                    + file
315                    + "\" - Package Owner=<4>\r\n\r\n");
316 
317       writer.write("Package=<5>\r\n{{{\r\n}}}\r\n\r\n");
318       writer.write("Package=<4>\r\n{{{\r\n");
319       if (dependsOn != null) {
320         for(Iterator iter = dependsOn.iterator(); iter.hasNext();) {
321             writer.write("    Begin Project Dependency\r\n");
322             writer.write("    Project_Dep_Name " + toProjectName(String.valueOf(iter.next())) + "\r\n");
323             writer.write("    End Project Dependency\r\n");
324         }
325       }
326       writer.write("}}}\r\n\r\n");
327 
328   }

330   private void writeWorkspace(final Writer writer,
331                               final ProjectDef project,
332                               final String projectName,
333                               final File dspFile) throws IOException {
334 
335       writer.write("Microsoft Developer Studio Workspace File, Format Version ");
336       writer.write(version);
337       writer.write("\r\n");
338       writer.write("# WARNING: DO NOT EDIT OR DELETE");
339       writer.write(" THIS WORKSPACE FILE!\r\n\r\n");
340 
341 	  writeComments(writer, project.getComments());
342 
343 
344       List dependencies = project.getDependencies();
345       List projectDeps = new ArrayList();
346       String basePath = dspFile.getParent();
347       for(Iterator iter = dependencies.iterator(); iter.hasNext();) {
348           DependencyDef dep = (DependencyDef) iter.next();
349           if (dep.getFile() != null) {
350             String projName = toProjectName(dep.getName());
351             projectDeps.add(projName);
352             String depProject = CUtil.toWindowsPath(
353                       CUtil.getRelativePath(basePath,
354                               new File(dep.getFile() + ".dsp")));
355             writeWorkspaceProject(writer, projName, depProject, dep.getDependsList());
356           }
357       }
358 
359       writeWorkspaceProject(writer, projectName, dspFile.getName(), projectDeps);
360 
361       writer.write("############################################");
362       writer.write("###################################\r\n\r\n");
363 
364 
365       writer.write("Global:\r\n\r\nPackage=<5>\r\n{{{\r\n}}}");
366       writer.write("\r\n\r\nPackage=<3>\r\n{{{\r\n}}}\r\n\r\n");
367 
368       writer.write("########################################");
369       writer.write("#######################################\r\n\r\n");
370 
371   }

373   /**
374    * Returns true if the file has an extension that appears in the group filter.
375    * @param filter String group filter
376    * @param candidate File file
377    * @return boolean true if member of group
378    */
379   private boolean isGroupMember(final String filter, final File candidate) {
380     String fileName = candidate.getName();
381     int lastDot = fileName.lastIndexOf('.');
382     if (lastDot >= 0 && lastDot < fileName.length() - 1) {
383       String extension =
384            ";" + fileName.substring(lastDot + 1).toLowerCase() + ";";
385       String semiFilter = ";" + filter + ";";
386       return semiFilter.indexOf(extension) >= 0;
387     }
388     return false;
389   }

391   /**
392    * Writes the entry for one source file in the project.
393    * @param writer Writer writer
394    * @param basePath String base path for project
395    * @param groupMember File project source file
396    * @throws IOException if error writing project file
397    */
398   private void writeSource(final Writer writer,
399                            final String basePath,
400                            final File groupMember)
401       throws IOException {
402     writer.write("# Begin Source File\r\n\r\nSOURCE=");
403     String relativePath = CUtil.getRelativePath(basePath,
404                                                 groupMember);
405     //
406     //  if relative path is just a name (hello.c) then
407     //    make it .\hello.c
408     if (!relativePath.startsWith(".")
409         && relativePath.indexOf(":") < 0
410         && !relativePath.startsWith("\\")) {
411       relativePath = ".\\" + relativePath;
412     }
413     writer.write(CUtil.toWindowsPath(relativePath));
414     writer.write("\r\n# End Source File\r\n");
415   }

417   /**
418    * Get alphabetized array of source files.
419    * @param sourceList list of source files
420    * @return File[] source files
421    */
422   private File[] getSources(final List sourceList) {
423     File[] sortedSources = new File[sourceList.size()];
424     sourceList.toArray(sortedSources);
425     Arrays.sort(sortedSources, new Comparator() {
426       public int compare(final Object o1, final Object o2) {
427         return ((File) o1).getName().compareTo(((File) o2).getName());
428       }
429     });
430     return sortedSources;
431   }

450     writer.write(projectName);
451     writer.write(".mak\".\r\n");
452     writer.write("!MESSAGE \r\n");
453     writer.write(
454         "!MESSAGE You can specify a configuration when running NMAKE\r\n");
455     writer.write(
456         "!MESSAGE by defining the macro CFG on the command line. ");
457     writer.write("For example:\r\n");
458     writer.write("!MESSAGE \r\n");
459     writer.write("!MESSAGE NMAKE /f \"");
460     writer.write(projectName);
461     writer.write(".mak\" CFG=\"");
462     writer.write(projectName);
463     writer.write(" - Win32 Debug\"\r\n");
464     writer.write("!MESSAGE \r\n");
465     writer.write("!MESSAGE Possible choices for configuration are:\r\n");
466     writer.write("!MESSAGE \r\n");

469     writer.write(MessageFormat.format(pattern, new Object[] { projectName, "Debug", targtype }));

475   /**
476    * Gets the first recognized compiler from the
477    * compilation targets.
478    * @param targets compilation targets
479    * @return representative (hopefully) compiler configuration
480    */
481   private CommandLineCompilerConfiguration
482       getBaseCompilerConfiguration(final Map targets) {
483     //
484     //   find first target with an DevStudio C compilation
485     //
486     CommandLineCompilerConfiguration compilerConfig;
487     //
488     //   get the first target and assume that it is representative
489     //
490     Iterator targetIter = targets.values().iterator();
491     while (targetIter.hasNext()) {
492       TargetInfo targetInfo = (TargetInfo) targetIter.next();
493       ProcessorConfiguration config = targetInfo.getConfiguration();
494       //
495       //   for the first cl compiler
496       //
497       if (config instanceof CommandLineCompilerConfiguration) {
498         compilerConfig = (CommandLineCompilerConfiguration) config;
499         if (compilerConfig.getCompiler() instanceof DevStudioCCompiler) {
500           return compilerConfig;
501         }
502       }
503     }
504     return null;
505   }

507   /**
508    * Writes compiler options.
509    * @param writer Writer writer
510    * @param isDebug true if debug.
511    * @param baseDir String base directory
512    * @param compilerConfig compiler configuration
513    * @throws IOException if error on writing project
514    */
515   private void writeCompileOptions(final Writer writer,
516                                    final boolean isDebug,
517                                    final String baseDir,
518                                    final CommandLineCompilerConfiguration
519                                    compilerConfig) throws IOException {
520     StringBuffer baseOptions = new StringBuffer(50);
521     baseOptions.append("# ADD BASE CPP");
522     StringBuffer options = new StringBuffer(50);
523     options.append("# ADD CPP");
524     File[] includePath = compilerConfig.getIncludePath();
525     for (int i = 0; i < includePath.length; i++) {
526       options.append(" /I \"");
527       String relPath = CUtil.getRelativePath(baseDir, includePath[i]);
528       options.append(CUtil.toWindowsPath(relPath));
529       options.append('"');
530     }
531     Hashtable optionMap = new Hashtable();
532 
533     if (isDebug) {
534         //
535         //   release options that should be mapped to debug counterparts
536         //
537         optionMap.put("/MT", "/MTd");
538         optionMap.put("/ML", "/MLd");
539         optionMap.put("/MD", "/MDd");
540         optionMap.put("/O2", "/Od");
541         optionMap.put("/O3", "/Od");
542     } else {
543         //
544         //   debug options that should be mapped to release counterparts
545         //
546         optionMap.put("/MTD", "/MT");
547         optionMap.put("/MLD", "/ML");
548         optionMap.put("/MDD", "/MD");
549         optionMap.put("/GM", "");
550         optionMap.put("/ZI", "");
551         optionMap.put("/OD", "/O2");
552         optionMap.put("/GZ", "");
553     }
554 
555 
556 
557     String[] preArgs = compilerConfig.getPreArguments();
558     for (int i = 0; i < preArgs.length; i++) {
559       if (preArgs[i].startsWith("/D")) {
560         options.append(" /D ");
561         baseOptions.append(" /D ");
562         String body = preArgs[i].substring(2);
563         if (preArgs[i].indexOf('=') >= 0) {
564           options.append(body);
565           baseOptions.append(body);
566         } else {
567           StringBuffer buf = new StringBuffer("\"");
568           if ("NDEBUG".equals(body) || "_DEBUG".equals(body)) {
569             if (isDebug) {
570                 buf.append("_DEBUG");
571             } else {
572                 buf.append("NDEBUG");
573             }
574           } else {
575             buf.append(body);
576           }
577           buf.append("\"");
578           options.append(buf);
579           baseOptions.append(buf);
580         }
581       } else if (!preArgs[i].startsWith("/I")) {
582         String option = preArgs[i];
583         String key = option.toUpperCase(Locale.US);
584         if (optionMap.containsKey(key)) {
585             option = optionMap.get(key).toString();
586         }
587         options.append(" ");
588         options.append(option);
589         baseOptions.append(" ");
590         baseOptions.append(option);
591       }
592     }
593     baseOptions.append("\r\n");
594     options.append("\r\n");
595     writer.write(baseOptions.toString());
596     writer.write(options.toString());
597 
598   }

604   /**
605    * Writes link options.
606    * @param writer Writer writer
607    * @param basePath String base path
608    * @param dependencies project dependencies, used to suppress explicit linking.
609    * @param linkTarget TargetInfo link target
610    * @param targets Hashtable all targets
611    * @throws IOException if unable to write to project file
612    */
613   private void writeLinkOptions(final Writer writer,
614                                 final boolean isDebug,
615                                 final List dependencies,
616                                 final String basePath,
617                                 final TargetInfo linkTarget,
618                                 final Map targets) throws IOException {
619 
620     StringBuffer baseOptions = new StringBuffer(100);
621     StringBuffer options = new StringBuffer(100);
622     baseOptions.append("# ADD BASE LINK32");
623     options.append("# ADD LINK32");
624 
625     ProcessorConfiguration config = linkTarget.getConfiguration();
626     if (config instanceof CommandLineLinkerConfiguration) {
627       CommandLineLinkerConfiguration linkConfig =
628           (CommandLineLinkerConfiguration) config;
629 
630       File[] linkSources = linkTarget.getAllSources();
631       for (int i = 0; i < linkSources.length; i++) {
632         //
633         //   if file was not compiled or otherwise generated
634         //
635         if (targets.get(linkSources[i].getName()) == null) {
636           //
637           //   if source appears to be a system library or object file
638           //      just output the name of the file (advapi.lib for example)
639           //      otherwise construct a relative path.
640           //
641           String relPath = linkSources[i].getName();
642           //
643           //   check if file comes from a project dependency
644           //       if it does it should not be explicitly linked
645           boolean fromDependency = false;
646           if (relPath.indexOf(".") > 0) {
647               String baseName = relPath.substring(0, relPath.indexOf("."));
648               for(Iterator iter = dependencies.iterator(); iter.hasNext(); ) {
649                 DependencyDef depend = (DependencyDef) iter.next();
650                 if (baseName.compareToIgnoreCase(depend.getName()) == 0) {
651                     fromDependency = true;
652                 }
653             }
654           }
655           if (!fromDependency) {
656             if (!CUtil.isSystemPath(linkSources[i])) {
657               relPath = CUtil.getRelativePath(basePath, linkSources[i]);
658             }
659             //
660             //   if path has an embedded space then
661             //      must quote
662             if (relPath.indexOf(' ') > 0) {
663                 options.append(" \"");
664                 options.append(CUtil.toWindowsPath(relPath));
665                 options.append("\"");
666             } else {
667                 options.append(' ');
668                 options.append(CUtil.toWindowsPath(relPath));
669             }
670            }
671         }
672       }
673       String[] preArgs = linkConfig.getPreArguments();
674       for (int i = 0; i < preArgs.length; i++) {
675         if (isDebug || !preArgs[i].equals("/DEBUG")) {
676             options.append(' ');
677             options.append(preArgs[i]);
678             baseOptions.append(' ');
679             baseOptions.append(preArgs[i]);
680         }
681       }
682       String[] endArgs = linkConfig.getEndArguments();
683       for (int i = 0; i < endArgs.length; i++) {
684         options.append(' ');
685         options.append(endArgs[i]);
686         baseOptions.append(' ');
687         baseOptions.append(endArgs[i]);
688       }
689     }
690     baseOptions.append("\r\n");
691     options.append("\r\n");
692     writer.write(baseOptions.toString());
693     writer.write(options.toString());
694   }
