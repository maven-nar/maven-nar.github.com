30  /**
31   * An abstract base class for compilers that are basically command line
32   * compatible with Microsoft(r) C/C++ Optimizing Compiler
33   *
34   * @author Curt Arnold
35   */
36  public abstract class DevStudioCompatibleCCompiler
37          extends
38              PrecompilingCommandLineCCompiler {
39      private static String[] mflags = new String[]{
40      //
41              //   first four are single-threaded
42              //      (runtime=static,debug=false), (..,debug=true),
43              //      (runtime=dynamic,debug=true), (..,debug=false), (not supported)
44              //    next four are multi-threaded, same sequence
45              "/ML", "/MLd", null, null, "/MT", "/MTd", "/MD", "/MDd"};
46      protected DevStudioCompatibleCCompiler(String command,
47              String identifierArg, boolean newEnvironment, Environment env) {
48          super(command, identifierArg, new String[]{".c", ".cc", ".cpp", ".cxx",
49                  ".c++"}, new String[]{".h", ".hpp", ".inl"}, ".obj", false,
50                  null, newEnvironment, env);
51      }
52      protected void addImpliedArgs(final Vector args,
53                      final boolean debug,
54              final boolean multithreaded,
55                          final boolean exceptions,
56                          final LinkType linkType,
57                          final Boolean rtti,
58                          final OptimizationEnum optimization) {
59          args.addElement("/c");
60          args.addElement("/nologo");
61          if (exceptions) {
62              //   changed to eliminate warning on VC 2005, should support VC 6 and later
63              //   use /GX to support VC5 - 2005 (with warning)
64              args.addElement("/EHsc");
65          }
66          int mindex = 0;
67          if (multithreaded) {
68              mindex += 4;
69          }
70          boolean staticRuntime = linkType.isStaticRuntime();
71          if (!staticRuntime) {
72              mindex += 2;
73          }
74          if (debug) {
75              mindex += 1;
76              addDebugSwitch(args);
77          } else {
78                  if (optimization != null) {
79                     if (optimization.isSize()) {
80                       args.addElement("/O1");
81                     }
82                     if (optimization.isSpeed()) {
83                       args.addElement("/O2");
84                     }
85                  }
86              args.addElement("/DNDEBUG");
87          }
88          String mflag = mflags[mindex];
89          if (mflag == null) {
90              throw new BuildException(
91                      "multithread='false' and runtime='dynamic' not supported");
92          }
93          args.addElement(mflag);
94          if (rtti != null && rtti.booleanValue()) {
95                  args.addElement("/GR");
96           } else {
97                  // added by Darren Sargent, 21Mar2008 -- /GR is default so need
98                  // /GR- to disable it
99                  args.addElement("/GR-");
100         }
101     }
102     protected void addDebugSwitch(Vector args) {
103         args.addElement("/Zi");
104         args.addElement("/Od");
105         args.addElement("/RTC1");
106         args.addElement("/D_DEBUG");
107     }
108     protected void addWarningSwitch(Vector args, int level) {
109         DevStudioProcessor.addWarningSwitch(args, level);
110     }
111     protected CompilerConfiguration createPrecompileGeneratingConfig(
112             CommandLineCompilerConfiguration baseConfig, File prototype,
113             String lastInclude) {
114         String[] additionalArgs = new String[]{
115                 "/Fp" + CUtil.getBasename(prototype) + ".pch", "/Yc"};
116         // FREEHEP FIXME we may need /Yd here, but only in debug mode, how do we find out?
117         return new CommandLineCompilerConfiguration(baseConfig, additionalArgs,
118                 null, true);
119     }
120     protected CompilerConfiguration createPrecompileUsingConfig(
121             CommandLineCompilerConfiguration baseConfig, File prototype,
122             String lastInclude, String[] exceptFiles) {
123         String[] additionalArgs = new String[]{
124                 "/Fp" + CUtil.getBasename(prototype) + ".pch",
125                 "/Yu" + lastInclude};
126 
127         return new CommandLineCompilerConfiguration(baseConfig, additionalArgs,
128                 exceptFiles, false);
129     }
130     protected void getDefineSwitch(StringBuffer buffer, String define,
131             String value) {
132         DevStudioProcessor.getDefineSwitch(buffer, define, value);
133     }
134     protected File[] getEnvironmentIncludePath() {
135         return CUtil.getPathFromEnvironment("INCLUDE", ";");
136     }
137     protected String getIncludeDirSwitch(String includeDir) {
138         return DevStudioProcessor.getIncludeDirSwitch(includeDir);
139     }
140     protected void getUndefineSwitch(StringBuffer buffer, String define) {
141         DevStudioProcessor.getUndefineSwitch(buffer, define);
142     }
143 }
