24  /**
25   * Adapter for the Microsoft(r) C/C++ Optimizing Compiler
26   * 
27   * @author Adam Murdoch
28   */
29  public final class DevStudioCCompiler extends DevStudioCompatibleCCompiler {
30      private static final DevStudioCCompiler instance = new DevStudioCCompiler(
31              "cl", false, null);
32      public static DevStudioCCompiler getInstance() {
33          return instance;
34      }
35      private DevStudioCCompiler(String command, boolean newEnvironment,
36              Environment env) {
37          super(command, "/bogus", newEnvironment, env);
38      }
39      public Processor changeEnvironment(boolean newEnvironment, Environment env) {
40          if (newEnvironment || env != null) {
41              return new DevStudioCCompiler(getCommand(), newEnvironment, env);
42          }
43          return this;
44      }
45      public Linker getLinker(LinkType type) {
46          return DevStudioLinker.getInstance().getLinker(type);
47      }
48      public int getMaximumCommandLength() {
49  // FREEHEP stay on safe side
50          return 32000; // 32767;
51      }
52  }
