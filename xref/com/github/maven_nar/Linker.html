188     public final String getName( NarProperties properties, String prefix )
189         throws MojoFailureException, MojoExecutionException
190     {
191         if ( ( name == null ) && ( properties != null ) && ( prefix != null ) )
192         {
193             name = properties.getProperty( prefix + "linker" );
194         }
195         if ( name == null )
196         {
197             throw new MojoExecutionException( "NAR: One of two things may be wrong here:\n\n"
198                 + "1. <Name> tag is missing inside the <Linker> tag of your NAR configuration\n\n"
199                 + "2. no linker is defined in the aol.properties file for '" + prefix + "linker'\n" );
200         }
201         return name;
202     }

204     public final String getVersion() 
205         throws MojoFailureException, MojoExecutionException
206     {
207         if ( name == null )
208         {
209             throw new MojoFailureException( "Cannot deduce linker version if name is null" );
210         }
211 
212         String version = null;
213 
214         TextStream out = new StringTextStream();
215         TextStream err = new StringTextStream();
216         TextStream dbg = new StringTextStream();
217 
218         if ( name.equals( "g++" ) || name.equals( "gcc" ) )
219         {
220             NarUtil.runCommand( "gcc", new String[] { "--version" }, null, null, out, err, dbg, log );
221             Pattern p = Pattern.compile( "\\d+\\.\\d+\\.\\d+" );
222             Matcher m = p.matcher( out.toString() );
223             if ( m.find() )
224             {
225                 version = m.group( 0 );
226             }
227         }
228         else if ( name.equals( "msvc" ) )
229         {
230             NarUtil.runCommand( "link", new String[] { "/?" }, null, null, out, err, dbg, log, true );
231             Pattern p = Pattern.compile( "\\d+\\.\\d+\\.\\d+(\\.\\d+)?" );
232             Matcher m = p.matcher( out.toString() );
233             if ( m.find() )
234             {
235                 version = m.group( 0 );
236             }
237         }
238         else if ( name.equals( "icc" ) || name.equals( "icpc" ) )
239         {
240             NarUtil.runCommand( "icc", new String[] { "--version" }, null, null, out, err, dbg, log );
241             Pattern p = Pattern.compile( "\\d+\\.\\d+" );
242             Matcher m = p.matcher( out.toString() );
243             if ( m.find() )
244             {
245                 version = m.group( 0 );
246             }
247         }
248         else if ( name.equals( "icl" ) )
249         {
250             NarUtil.runCommand( "icl", new String[] { "/QV" }, null, null, out, err, dbg, log );
251             Pattern p = Pattern.compile( "\\d+\\.\\d+" );
252             Matcher m = p.matcher( err.toString() );
253             if ( m.find() )
254             {
255                 version = m.group( 0 );
256             }
257         }
258         else if ( name.equals( "CC" ) )
259         {
260         	NarUtil.runCommand( "CC", new String[] { "-V" }, null, null, out, err, dbg, log );
261         	Pattern p = Pattern.compile( "\\d+\\.d+" );
262         	Matcher m = p.matcher( err.toString() );
263         	if ( m.find() )
264         	{ 
265         		version = m.group( 0 ); 
266         	}
267         }
268         else
269         {
270             throw new MojoFailureException( "Cannot find version number for linker '" + name + "'" );
271         }
272         
273         if (version == null) {
274         	throw new MojoFailureException( "Cannot deduce version number from: " + out.toString() );
275         }
276         return version;
277     }

299     public final LinkerDef getLinker( AbstractCompileMojo mojo, Project antProject, String os, String prefix,
300                                       String type )
301         throws MojoFailureException, MojoExecutionException
302     {
303         if ( name == null )
304         {
305             throw new MojoFailureException( "NAR: Please specify a <Name> as part of <Linker>" );
306         }
307 
308         LinkerDef linker = new LinkerDef();
309         linker.setProject( antProject );
310         LinkerEnum linkerEnum = new LinkerEnum();
311         linkerEnum.setValue( name );
312         linker.setName( linkerEnum );
313 
314         // tool path
315         if ( toolPath != null )
316         {
317             linker.setToolPath( toolPath );
318         }
319 
320         // incremental, map
321         linker.setIncremental( incremental );
322         linker.setMap( map );
323 
324         // Add definitions (Window only)
325         if ( os.equals( OS.WINDOWS ) && getName( null, null ).equals( "msvc" ) && ( type.equals( Library.SHARED ) || type.equals( Library.JNI ) ) )
326         {
327             Set defs = new HashSet();
328             try
329             {
330                 if ( mojo.getC() != null )
331                 {
332                     List cSrcDirs = mojo.getC().getSourceDirectories();
333                     for ( Iterator i = cSrcDirs.iterator(); i.hasNext(); )
334                     {
335                         File dir = (File) i.next();
336                         if ( dir.exists() )
337                         {
338                             defs.addAll( FileUtils.getFiles( dir, "**/*.def", null ) );
339                         }
340                     }
341                 }
342             }
343             catch ( IOException e )
344             {
345             }
346             try
347             {
348                 if ( mojo.getCpp() != null )
349                 {
350                     List cppSrcDirs = mojo.getCpp().getSourceDirectories();
351                     for ( Iterator i = cppSrcDirs.iterator(); i.hasNext(); )
352                     {
353                         File dir = (File) i.next();
354                         if ( dir.exists() )
355                         {
356                             defs.addAll( FileUtils.getFiles( dir, "**/*.def", null ) );
357                         }
358                     }
359                 }
360             }
361             catch ( IOException e )
362             {
363             }
364             try
365             {
366                 if ( mojo.getFortran() != null )
367                 {
368                     List fortranSrcDirs = mojo.getFortran().getSourceDirectories();
369                     for ( Iterator i = fortranSrcDirs.iterator(); i.hasNext(); )
370                     {
371                         File dir = (File) i.next();
372                         if ( dir.exists() )
373                         {
374                             defs.addAll( FileUtils.getFiles( dir, "**/*.def", null ) );
375                         }
376                     }
377                 }
378             }
379             catch ( IOException e )
380             {
381             }
382 
383             for ( Iterator i = defs.iterator(); i.hasNext(); )
384             {
385                 LinkerArgument arg = new LinkerArgument();
386                 arg.setValue( "/def:" + i.next() );
387                 linker.addConfiguredLinkerArg( arg );
388             }
389         }
390 
391         // FIXME, this should be done in CPPTasks at some point, and may not be necessary, but was for VS 2010 beta 2
392         if ( os.equals( OS.WINDOWS ) && getName( null, null ).equals( "msvc" ) && !getVersion().startsWith( "6." ) )
393         {
394             LinkerArgument arg = new LinkerArgument();
395             arg.setValue( "/MANIFEST" );
396             linker.addConfiguredLinkerArg( arg );
397         }
398 
399         // Add options to linker
400         if ( options != null )
401         {
402             for ( Iterator i = options.iterator(); i.hasNext(); )
403             {
404                 LinkerArgument arg = new LinkerArgument();
405                 arg.setValue( (String) i.next() );
406                 linker.addConfiguredLinkerArg( arg );
407             }
408         }
409 
410         if ( optionSet != null )
411         {
412 
413             String[] opts = optionSet.split( "\\s" );
414 
415             for ( int i = 0; i < opts.length; i++ )
416             {
417 
418                 LinkerArgument arg = new LinkerArgument();
419 
420                 arg.setValue( opts[i] );
421                 linker.addConfiguredLinkerArg( arg );
422             }
423         }
424 
425         if ( !clearDefaultOptions )
426         {
427             String option = NarProperties.getInstance(mojo.getMavenProject()).getProperty( prefix + "options" );
428             if ( option != null )
429             {
430                 String[] opt = option.split( " " );
431                 for ( int i = 0; i < opt.length; i++ )
432                 {
433                     LinkerArgument arg = new LinkerArgument();
434                     arg.setValue( opt[i] );
435                     linker.addConfiguredLinkerArg( arg );
436                 }
437             }
438         }
439 
440         // record the preference for nar dependency library link order
441         if ( narDependencyLibOrder != null )
442         {
443 
444             List libOrder = new LinkedList();
445 
446             String[] lib = narDependencyLibOrder.split( "," );
447 
448             for ( int i = 0; i < lib.length; i++ )
449             {
450                 libOrder.add( lib[i].trim() );
451             }
452 
453             mojo.setDependencyLibOrder( libOrder );
454         }
455 
456         // Add Libraries to linker
457         if ( ( libs != null ) || ( libSet != null ) )
458         {
459 
460             if ( libs != null )
461             {
462 
463                 for ( Iterator i = libs.iterator(); i.hasNext(); )
464                 {
465 
466                     Lib lib = (Lib) i.next();
467                     lib.addLibSet( mojo, linker, antProject );
468                 }
469             }
470 
471             if ( libSet != null )
472             {
473                 addLibraries( libSet, linker, antProject, false );
474             }
475         }
476         else
477         {
478 
479             String libsList = NarProperties.getInstance(mojo.getMavenProject()).getProperty( prefix + "libs" );
480 
481             addLibraries( libsList, linker, antProject, false );
482         }
483 
484         // Add System Libraries to linker
485         if ( ( sysLibs != null ) || ( sysLibSet != null ) )
486         {
487 
488             if ( sysLibs != null )
489             {
490 
491                 for ( Iterator i = sysLibs.iterator(); i.hasNext(); )
492                 {
493 
494                     SysLib sysLib = (SysLib) i.next();
495                     linker.addSyslibset( sysLib.getSysLibSet( antProject ) );
496                 }
497             }
498 
499             if ( sysLibSet != null )
500             {
501                 addLibraries( sysLibSet, linker, antProject, true );
502             }
503         }
504         else
505         {
506 
507             String sysLibsList = NarProperties.getInstance(mojo.getMavenProject()).getProperty( prefix + "sysLibs" );
508 
509             addLibraries( sysLibsList, linker, antProject, true );
510         }
511 
512         return linker;
513     }

515     private void addLibraries( String libraryList, LinkerDef linker, Project antProject, boolean isSystem )
516     {
517 
518         if ( libraryList == null )
519         {
520             return;
521         }
522 
523         String[] lib = libraryList.split( "," );
524 
525         for ( int i = 0; i < lib.length; i++ )
526         {
527 
528             String[] libInfo = lib[i].trim().split( ":", 3 );
529 
530             LibrarySet librarySet = new LibrarySet();
531 
532             if ( isSystem )
533             {
534                 librarySet = new SystemLibrarySet();
535             }
536 
537             librarySet.setProject( antProject );
538             librarySet.setLibs( new CUtil.StringArrayBuilder( libInfo[0] ) );
539 
540             if ( libInfo.length > 1 )
541             {
542 
543                 LibraryTypeEnum libType = new LibraryTypeEnum();
544 
545                 libType.setValue( libInfo[1] );
546                 librarySet.setType( libType );
547 
548                 if ( !isSystem && ( libInfo.length > 2 ) )
549                 {
550                     librarySet.setDir( new File( libInfo[2] ) );
551                 }
552             }
553 
554             if ( !isSystem )
555             {
556                 linker.addLibset( librarySet );
557             }
558             else
559             {
560                 linker.addSyslibset( (SystemLibrarySet) librarySet );
561             }
562         }
563     }
