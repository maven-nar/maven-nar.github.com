97    public void configureCCTask(final CCTask task) throws MojoExecutionException {
98      if (OS.WINDOWS.equals(mojo.getOS())
99            && "msvc".equalsIgnoreCase(mojo.getLinker().getName())) {
100       addIncludePath(task, this.home, "VC/include");
101       addIncludePath(task, this.home, "VC/atlmfc/include");
102       if (compareVersion(this.windowsSdkVersion, "7.1A") <= 0) {
103         addIncludePath(task, this.windowsSdkHome, "include");
104       } else {
105         for (File sdkInclude : sdkIncludes)
106           addIncludePathToTask(task, sdkInclude);
107 
108       }
109       task.addEnv(getPathVariable());
110       // TODO: supporting running with clean environment - addEnv sets
111       // newEnvironemnt by default
112       // task.setNewenvironment(false);
113       Variable envVariable = new Variable();
114       // cl needs SystemRoot env var set, otherwise D8037 is raised (bogus
115       // message)
116       // - https://msdn.microsoft.com/en-us/library/bb385201.aspx
117       // -
118       // http://stackoverflow.com/questions/10560779/cl-exe-when-launched-via-createprocess-does-not-seem-to-have-write-permissions
119       envVariable.setKey("SystemRoot");
120       envVariable.setValue(this.windowsHome.getAbsolutePath());
121       task.addEnv(envVariable);
122       // cl needs TMP otherwise D8050 is raised c1xx.dll
123       envVariable = new Variable();
124       envVariable.setKey("TMP");
125       envVariable.setValue(getTempPath());
126       task.addEnv(envVariable);
127     }

147       }
148       // 6 lib ?+ lib/x86 or lib/x64
149       if (compareVersion(this.windowsSdkVersion, "8.0") < 0) {
150         if ("x86".equals(arch)) {
151           linker.addLibraryDirectory(this.windowsSdkHome, "lib");
152         } else {
153           linker.addLibraryDirectory(this.windowsSdkHome, "lib/" + sdkArch);
154         }
155       } else
156         for (File sdkLib : sdkLibs)
157           linker.addLibraryDirectory(sdkLib, sdkArch);

204   private void initPath() throws MojoExecutionException {
205     final String mojoArchitecture = this.mojo.getArchitecture();
206     final String osArchitecture = NarUtil.getArchitecture(null);
207     // 32 bit build on 64 bit OS can be built with 32 bit tool, or 64 bit tool
208     // in amd64_x86 - currently defaulting to prefer 64 bit tools - match os
209     final boolean matchMojo = false;
210     // TODO: toolset architecture
211     // match os - os x86 mojo(x86 / x86_amd64); os x64 mojo(amd64_x86 / amd64);
212     // 32bit - force 32 on 64bit mojo(x86 / x86_amd64)
213     // match mojo - os x86 is as above; os x64 mojo (x86 / amd64)
214 
215     // Cross tools first if necessary, platform tools second, more generic tools
216     // later
217     if (!osArchitecture.equals(mojoArchitecture) && !matchMojo) {
218       if (!addPath(this.home, "VC/bin/" + osArchitecture + "_" + mojoArchitecture)) {
219         throw new MojoExecutionException("Unable to find compiler for architecture " + mojoArchitecture + ".\n"
220             + new File(this.home, "VC/bin/" + osArchitecture + "_" + mojoArchitecture));
221       }
222       toolPathLinker = new File(this.home, "VC/bin/" + osArchitecture + "_" + mojoArchitecture).getAbsolutePath();
223     }
224     if (null == toolPathLinker) {
225       if ("amd64".equals(mojoArchitecture))
226         toolPathLinker = new File(this.home, "VC/bin/amd64").getAbsolutePath();
227       else
228         toolPathLinker = new File(this.home, "VC/bin").getAbsolutePath();
229     }
230     if ("amd64".equals(osArchitecture) && !matchMojo) {
231       addPath(this.home, "VC/bin/amd64");
232     } else {
233       addPath(this.home, "VC/bin");
234     }
235     addPath(this.home, "VC/VCPackages");
236     addPath(this.home, "Common7/Tools");
237     addPath(this.home, "Common7/IDE");
238 
239     // 64 bit tools if present are preferred
240     if (compareVersion(this.windowsSdkVersion, "7.1A") <= 0) {
241       if ("amd64".equals(osArchitecture) && !matchMojo) {
242         addPath(this.windowsSdkHome, "bin/x64");
243       }
244       addPath(this.windowsSdkHome, "bin");
245     } else {
246       if ("amd64".equals(osArchitecture) && !matchMojo) {
247         addPath(this.windowsSdkHome, "bin/x64");
248       }
249       addPath(this.windowsSdkHome, "bin/x86");
250     }
251     if ("amd64".equals(mojoArchitecture)) {
252       toolPathWindowsSDK = new File(this.windowsSdkHome, "bin/x64").getAbsolutePath();
253     } else if (compareVersion(this.windowsSdkVersion, "7.1A") <= 0) {
254       toolPathWindowsSDK = new File(this.windowsSdkHome, "bin").getAbsolutePath();
255     } else {
256       toolPathWindowsSDK = new File(this.windowsSdkHome, "bin/x86").getAbsolutePath();
257     }
258 
259     // clearing the path, add back the windows system folders
260     addPath(this.windowsHome, "System32");
261     addPath(this.windowsHome, "");
262     addPath(this.windowsHome, "System32/wbem");
263   }

265   private void initVisualStudio() throws MojoFailureException, MojoExecutionException {
266     mojo.getLog().debug(" -- Searching for usable VisualStudio ");
267     if (this.version != null && this.version.trim().length() > 1) {
268       String internalVersion;
269       Pattern r = Pattern.compile("(\\d+)\\.*(\\d)");
270       Matcher matcher = r.matcher(this.version);
271       if (matcher.find()) {
272         internalVersion = matcher.group(1) + matcher.group(2);
273         this.version = matcher.group(1) + "." + matcher.group(2);
274       } else {
275         throw new MojoExecutionException("msvc.version must be the internal version in the form 10.0 or 120");
276       }
277       if (this.home == null) {
278         final String commontToolsVar = System.getenv("VS" + internalVersion + "COMNTOOLS");
279         if (commontToolsVar != null && commontToolsVar.trim().length() > 0) {
280           final File commonToolsDirectory = new File(commontToolsVar);
281           if (commonToolsDirectory.exists()) {
282             this.home = commonToolsDirectory.getParentFile().getParentFile();
283           }
284         }
285         // TODO: else Registry might be more reliable but adds dependency to be
286         // able to acccess - HKLM\SOFTWARE\Microsoft\Visual Studio\Major.Minor:InstallDir
287       }
288       mojo.getLog()
289           .debug(String.format(" VisualStudio %1s (%2s) found %3s ", this.version, internalVersion, this.home));
290     } else {
291       this.version = "";
292       for (final Entry<String, String> entry : System.getenv().entrySet()) {
293         final String key = entry.getKey();
294         final String value = entry.getValue();
295         final Pattern versionPattern = Pattern.compile("VS(\\d+)(\\d)COMNTOOLS");
296         final Matcher matcher = versionPattern.matcher(key);
297         if (matcher.matches()) {
298           final String version = matcher.group(1) + "." + matcher.group(2);
299           if (version.compareTo(this.version) > 0) {
300             final File commonToolsDirectory = new File(value);
301             if (commonToolsDirectory.exists()) {
302               this.version = version;
303               this.home = commonToolsDirectory.getParentFile().getParentFile();
304               mojo.getLog().debug(
305                   String.format(" VisualStudio %1s (%2s) found %3s ", this.version,
306                       matcher.group(1) + matcher.group(2), this.home));
307             }
308           }
309         }
310       }
311       if (this.version.length() == 0) {
312         final TextStream out = new StringTextStream();
313         final TextStream err = new StringTextStream();
314         final TextStream dbg = new StringTextStream();
315 
316         NarUtil.runCommand("link", new String[] {
317           "/?"
318         }, null, null, out, err, dbg, null, true);
319         final Pattern p = Pattern.compile("(\\d+\\.\\d+)\\.\\d+(\\.\\d+)?");
320         final Matcher m = p.matcher(out.toString());
321         if (m.find()) {
322           this.version = m.group(1);
323           mojo.getLog().debug(
324               String.format(" VisualStudio Not found but link runs and reports version %1s (%2s)", this.version,
325                   m.group(0)));
326         } else {
327           throw new MojoExecutionException(
328               "msvc.version not specified and no VS<Version>COMNTOOLS environment variable can be found");
329         }
330       }
331     }

334   private final Comparator<File> versionComparator = new Comparator<File>() {
335     @Override
336     public int compare(File o1, File o2) {
337       // will be sorted smallest first, so we need to invert the order of
338       // the objects
339       String firstDir = o2.getName(), secondDir = o1.getName();
340       if (firstDir.charAt(0) == 'v') { // remove 'v' and 'A' at the end
341         firstDir = firstDir.substring(1, firstDir.length() - 1);
342         secondDir = secondDir.substring(1, secondDir.length() - 1);
343       }
344       // impossible that two dirs are the same
345       String[] firstVersionString = firstDir.split("\\."), secondVersionString = secondDir.split("\\.");
346 
347       int maxIdx = Math.min(firstVersionString.length, secondVersionString.length);
348       int deltaVer;
349       try {
350         for (int i = 0; i < maxIdx; i++)
351           if ((deltaVer = Integer.parseInt(firstVersionString[i]) - Integer.parseInt(secondVersionString[i])) != 0)
352             return deltaVer;
353 
354       } catch (NumberFormatException e) {
355         return firstDir.compareTo(secondDir);
356       }
357       if (firstVersionString.length > maxIdx) // 10.0.150 > 10.0
358         return 1;
359       else if (secondVersionString.length > maxIdx) // 10.0 < 10.0.150
360         return -1;
361       return 0; // impossible that they are the same
362     }

365   private void initWindowsSdk() throws MojoExecutionException {
366     if(this.windowsSdkVersion != null && this.windowsSdkVersion.trim().equals(""))
367       this.windowsSdkVersion = null;
368 
369     mojo.getLog().debug(" -- Searching for usable WindowSDK ");
370     // newer first: 10 -> 8.1 -> 8.0 -> 7.1 and look for libs specified
371     for (final File directory : Arrays.asList(
372             new File("C:/Program Files (x86)/Windows Kits"),
373             new File("C:/Program Files (x86)/Microsoft SDKs/Windows"),
374             new File("C:/Program Files/Windows Kits"),
375             new File("C:/Program Files/Microsoft SDKs/Windows") )) {
376       if (directory.exists()) {
377         final File[] kitDirectories = directory.listFiles();
378         Arrays.sort(kitDirectories, versionComparator);
379         if (kitDirectories != null) {
380           for (final File kitDirectory : kitDirectories) {
381 
382             if (new File(kitDirectory, "Include").exists()) {
383               // legacy SDK
384               String kitVersion = kitDirectory.getName();
385               if (kitVersion.charAt(0) == 'v') {
386                 kitVersion = kitVersion.substring(1);
387               }
388               if (this.windowsSdkVersion!=null && compareVersion(kitVersion,this.windowsSdkVersion)>0)
389                 continue; // skip versions higher than the previous version
390               mojo.getLog()
391                   .debug(String.format(" WindowSDK %1s found %2s", kitVersion, kitDirectory.getAbsolutePath()));
392               if (kitVersion.matches("\\d+\\.\\d+?[A-Z]?")) {
393                 // windows <= 8.1
394                 legacySDK(kitDirectory);
395               } else if (kitVersion.matches("\\d+?")) {
396                 // windows 10 SDK supports
397                 addNewSDKLibraries(kitDirectory);
398               }
399 	        }
400 	      }
401           if (libsRequired.size() == 0) // need it here to break out of the outer loop
402               break;
403 	    }
404 	  }
405     }
406     if (!foundSDK)
407       throw new MojoExecutionException("msvc.windowsSdkVersion not specified and versions cannot be found");
408     mojo.getLog().debug(String.format(" Using WindowSDK %1s found %2s", this.windowsSdkVersion, this.windowsSdkHome));
409   }

412     // multiple installs
413     List<File> kitVersionDirectories = Arrays.asList(new File(kitDirectory, "Include").listFiles());
414     Collections.sort(kitVersionDirectories,  versionComparator);
415     ListIterator<File> kitVersionDirectoriesIt = kitVersionDirectories.listIterator();
416     File kitVersionDirectory = null;
417     while (kitVersionDirectoriesIt.hasNext() && (kitVersionDirectory = kitVersionDirectoriesIt.next()) != null) {
418       if (new File(kitVersionDirectory, "ucrt").exists()) {
419         break;
420       }
421     }
422

424       String version = kitVersionDirectory.getName();
425       mojo.getLog().debug(String.format(" Latest Win %1s KitDir at %2s", kitVersionDirectory.getName(), kitVersionDirectory.getAbsolutePath()));
426       // add the libraries found:

501   public int compareVersion(Object o1, Object o2) {
502     String version1 = (String) o1;
503     String version2 = (String) o2;
504 
505     VersionTokenizer tokenizer1 = new VersionTokenizer(version1);
506     VersionTokenizer tokenizer2 = new VersionTokenizer(version2);
507 
508     int number1 = 0, number2 = 0;
509     String suffix1 = "", suffix2 = "";
510 
511     while (tokenizer1.MoveNext()) {
512       if (!tokenizer2.MoveNext()) {
513         do {
514           number1 = tokenizer1.getNumber();
515           suffix1 = tokenizer1.getSuffix();
516           if (number1 != 0 || suffix1.length() != 0) {
517             // Version one is longer than number two, and non-zero
518             return 1;
519           }
520         } while (tokenizer1.MoveNext());
521 
522         // Version one is longer than version two, but zero
523         return 0;
524       }
525 
526       number1 = tokenizer1.getNumber();
527       suffix1 = tokenizer1.getSuffix();
528       number2 = tokenizer2.getNumber();
529       suffix2 = tokenizer2.getSuffix();
530 
531       if (number1 < number2) {
532         // Number one is less than number two
533         return -1;
534       }
535       if (number1 > number2) {
536         // Number one is greater than number two
537         return 1;
538       }
539 
540       boolean empty1 = suffix1.length() == 0;
541       boolean empty2 = suffix2.length() == 0;
542 
543       if (empty1 && empty2)
544         continue; // No suffixes
545       if (empty1)
546         return 1; // First suffix is empty (1.2 > 1.2b)
547       if (empty2)
548         return -1; // Second suffix is empty (1.2a < 1.2)
549 
550       // Lexical comparison of suffixes
551       int result = suffix1.compareTo(suffix2);
552       if (result != 0)
553         return result;
554 
555     }
556     if (tokenizer2.MoveNext()) {
557       do {
558         number2 = tokenizer2.getNumber();
559         suffix2 = tokenizer2.getSuffix();
560         if (number2 != 0 || suffix2.length() != 0) {
561           // Version one is longer than version two, and non-zero
562           return -1;
563         }
564       } while (tokenizer2.MoveNext());
565 
566       // Version two is longer than version one, but zero
567       return 0;
568     }
569     return 0;
570   }

573   class VersionTokenizer {
574     private final String _versionString;
575     private final int _length;
576 
577     private int _position;
578     private int _number;
579     private String _suffix;
580     private boolean _hasValue;
581 
582     public int getNumber() {
583       return _number;
584     }
585 
586     public String getSuffix() {
587       return _suffix;
588     }
589 
590     public boolean hasValue() {
591       return _hasValue;
592     }
593 
594     public VersionTokenizer(String versionString) {
595       if (versionString == null)
596         throw new IllegalArgumentException("versionString is null");
597 
598       _versionString = versionString;
599       _length = versionString.length();
600     }
601 
602     public boolean MoveNext() {
603       _number = 0;
604       _suffix = "";
605       _hasValue = false;
606 
607       // No more characters
608       if (_position >= _length)
609         return false;
610 
611       _hasValue = true;
612 
613       while (_position < _length) {
614         char c = _versionString.charAt(_position);
615         if (c < '0' || c > '9')
616           break;
617         _number = _number * 10 + (c - '0');
618         _position++;
619       }
620 
621       int suffixStart = _position;
622 
623       while (_position < _length) {
624         char c = _versionString.charAt(_position);
625         if (c == '.')
626           break;
627         _position++;
628       }
629 
630       _suffix = _versionString.substring(suffixStart, _position);
631 
632       if (_position < _length)
633         _position++;
634 
635       return true;
636     }
