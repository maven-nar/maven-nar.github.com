116 	public final List<NarArtifact> getNarArtifacts()
117 			throws MojoExecutionException {
118 		List<NarArtifact> narDependencies = new LinkedList<NarArtifact>();
119 		for (Iterator i = getArtifacts().iterator(); i.hasNext();) {
120 			Artifact dependency = (Artifact) i.next();
121 			getLog().debug("Examining artifact for NarInfo: " + dependency);
122 
123 			NarInfo narInfo = getNarInfo(dependency);
124 			if (narInfo != null) {
125 				getLog().debug("    - added as NarDependency");
126 				narDependencies.add(new NarArtifact(dependency, narInfo));
127 			}
128 		}
129 		getLog().debug(
130 				"Dependencies contained " + narDependencies.size()
131 						+ " NAR artifacts.");
132 		return narDependencies;
133 	}

135 	public final NarInfo getNarInfo(Artifact dependency)
136 			throws MojoExecutionException {
137 		// FIXME reported to maven developer list, isSnapshot changes behaviour
138 		// of getBaseVersion, called in pathOf.
139 		dependency.isSnapshot();
140 
141         if (dependency.getFile().isDirectory()) {
142             getLog().debug("Dependency is not packaged: " + dependency.getFile());
143 
144             return new NarInfo(dependency.getGroupId(), dependency.getArtifactId(), dependency.getBaseVersion(),
145                     getLog(), dependency.getFile());
146         }
147 
148 		File file = new File(getLocalRepository().getBasedir(),
149 				getLocalRepository().pathOf(dependency));
150 		if (!file.exists()) {
151 			getLog().debug("Dependency nar file does not exist: " + file);
152 			return null;
153 		}
154 
155 		JarFile jar = null;
156 		try {
157 			jar = new JarFile(file);
158 			NarInfo info = new NarInfo(dependency.getGroupId(),
159 					dependency.getArtifactId(), dependency.getBaseVersion(),
160 					getLog());
161 			if (!info.exists(jar)) {
162 				getLog().debug(
163 						"Dependency nar file does not contain this artifact: "
164 								+ file);
165 				return null;
166 			}
167 			info.read(jar);
168 			return info;
169 		} catch (IOException e) {
170 			throw new MojoExecutionException("Error while reading " + file, e);
171 		} finally {
172 			if (jar != null) {
173 				try {
174 					jar.close();
175 				} catch (IOException e) {
176 					// ignore
177 				}
178 			}
179 		}
180 	}

218 	private List<AttachedNarArtifact> getAttachedNarArtifacts(
219 			NarArtifact dependency, AOL aol, String type)
220 			throws MojoExecutionException, MojoFailureException {
221 		getLog().debug(
222 				"GetNarDependencies for " + dependency + ", aol: " + aol
223 						+ ", type: " + type);
224 		List<AttachedNarArtifact> artifactList = new ArrayList<AttachedNarArtifact>();
225 		NarInfo narInfo = dependency.getNarInfo();
226 		String[] nars = narInfo.getAttachedNars(aol, type);
227 		// FIXME Move this to NarInfo....
228 		if (nars != null) {
229 			for (int j = 0; j < nars.length; j++) {
230 				getLog().debug("    Checking: " + nars[j]);
231 				if (nars[j].equals("")) {
232 					continue;
233 				}
234 				String[] nar = nars[j].split(":", 5);
235 				if (nar.length >= 4) {
236 					try {
237 						String groupId = nar[0].trim();
238 						String artifactId = nar[1].trim();
239 						String ext = nar[2].trim();
240 						String classifier = nar[3].trim();
241 						// translate for instance g++ to gcc...
242 						AOL aolString = narInfo.getAOL(aol);
243 						if (aolString != null) {
244 							classifier = NarUtil.replace("${aol}",
245 									aolString.toString(), classifier);
246 						}
247 						String version = nar.length >= 5 ? nar[4].trim()
248 								: dependency.getVersion();
249 						artifactList.add(new AttachedNarArtifact(groupId,
250 								artifactId, version, dependency.getScope(),
251 								ext, classifier, dependency.isOptional(),
252 								dependency.getFile()));
253 					} catch (InvalidVersionSpecificationException e) {
254 						throw new MojoExecutionException(
255 								"Error while reading nar file for dependency "
256 										+ dependency, e);
257 					}
258 				} else {
259 					getLog().warn(
260 							"nars property in " + dependency.getArtifactId()
261 									+ " contains invalid field: '" + nars[j]
262 					// + "' for type: " + type
263 					);
264 				}
265 			}
266 		}
267 		return artifactList;
268 	}

270 	public final void downloadAttachedNars(
271 			List<AttachedNarArtifact> dependencies)
272 			throws MojoExecutionException, MojoFailureException {
273 		getLog().debug( "Download for NarDependencies {" );
274 		for (Iterator<AttachedNarArtifact> i = dependencies.iterator(); i.hasNext();) {
275 			getLog().debug("  - " + (i.next()));
276 		}
277 		getLog().debug("}");
278 
279 		for (Iterator<AttachedNarArtifact> i = dependencies.iterator(); i.hasNext();) {
280 			Artifact dependency = (Artifact) i.next();
281 			try {
282 				getLog().debug("Resolving " + dependency);
283 				artifactResolver.resolve(dependency, remoteArtifactRepositories, getLocalRepository());
284 			} catch (ArtifactNotFoundException e) {
285 				String message = "nar not found " + dependency.getId();
286 				throw new MojoExecutionException(message, e);
287 			} catch (ArtifactResolutionException e) {
288 				String message = "nar cannot resolve " + dependency.getId();
289 				throw new MojoExecutionException(message, e);
290 			}
291 		}
292 	}

294     public final void unpackAttachedNars( List<AttachedNarArtifact> dependencies )
295         	throws MojoExecutionException, MojoFailureException
296         {
297         	File unpackDir = getUnpackDirectory();
298 
299     	    getLog().info( String.format( "Unpacking %1$d dependencies to %2$s", dependencies.size(), unpackDir) );
300 
301     	    for ( Iterator i = dependencies.iterator(); i.hasNext(); )
302     	    {
303     	    	AttachedNarArtifact dependency = (AttachedNarArtifact) i.next();
304     		    File file = getNarManager().getNarFile( dependency ); //dependency.getNarFile();
305     		    getLog().debug( String.format( "Unpack %1$s (%2$s) to %3$s", dependency, file, unpackDir) );
306     		
307     		    // TODO: each dependency may have it's own (earlier) version of layout - if it is unknown then we should report an error to update the nar package
308     		    // NarLayout layout = AbstractNarLayout.getLayout( "NarLayout21"/* TODO: dependency.getLayout() */, getLog() );
309     		    // we should then target the layout to match the layout for this nar which is the workspace we are in.
310     		    NarLayout layout = getLayout();
311     		    // TODO: the dependency may be specified against a different linker (version)?
312     		    // AOL aol = dependency.getClassifier();  Trim
313     		    layout.unpackNar(unpackDir, archiverManager, file, getOS(), getLinker().getName(), getAOL() );            
314     	    }
315         }
