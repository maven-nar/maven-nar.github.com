252     private List<File> getSourceDirectories( String type )
253     {
254         List<File> sourceDirectories = new ArrayList<File>();
255         File baseDir = mojo.getMavenProject().getBasedir();
256 
257         if ( type.equals( TEST ) )
258         {
259             if ( testSourceDirectory == null )
260             {
261                 testSourceDirectory = new File( baseDir, "/src/test" );
262             }
263             if ( testSourceDirectory.exists() )
264             {
265                 sourceDirectories.add( testSourceDirectory );
266             }
267 
268             for ( Iterator i = mojo.getMavenProject().getTestCompileSourceRoots().iterator(); i.hasNext(); )
269             {
270                 File extraTestSourceDirectory = new File( (String) i.next() );
271                 if ( extraTestSourceDirectory.exists() )
272                 {
273                     sourceDirectories.add( extraTestSourceDirectory );
274                 }
275             }
276         }
277         else
278         {
279             if ( sourceDirectory == null )
280             {
281                 sourceDirectory = new File( baseDir, "src/main" );
282             }
283             if ( sourceDirectory.exists() )
284             {
285                 sourceDirectories.add( sourceDirectory );
286             }
287 
288             for ( Iterator i = mojo.getMavenProject().getCompileSourceRoots().iterator(); i.hasNext(); )
289             {
290                 File extraSourceDirectory = new File( (String) i.next() );
291                 if ( extraSourceDirectory.exists() )
292                 {
293                     sourceDirectories.add( extraSourceDirectory );
294                 }
295             }
296         }
297 
298         if ( mojo.getLog().isDebugEnabled() )
299         {
300             for ( Iterator<File> i = sourceDirectories.iterator(); i.hasNext(); )
301             {
302                 mojo.getLog().debug( "Added to sourceDirectory: " + i.next().getPath() );
303             }
304         }
305         return sourceDirectories;
306     }

308     protected final List<IncludePath> getIncludePaths( String type )
309     {
310         List<IncludePath> includeList = type.equals( TEST ) ? testIncludePaths : includePaths;
311 
312         if ( includeList != null && includeList.size() != 0 )
313             return includeList;
314 
315         includeList = new ArrayList<IncludePath>();
316         for ( Iterator<File> i = getSourceDirectories( type ).iterator(); i.hasNext(); )
317         {
318             //VR 20100318 only add include directories that exist - we now fail the build fast if an include directory does not exist
319             File file = new File( (File) i.next(), "include" );
320             if ( file.isDirectory() ) {
321                 IncludePath includePath = new IncludePath();
322                 includePath.setPath( file.getPath() );
323                 includeList.add( includePath );
324             }
325         }
326         return includeList;
327     }

335     protected final Set<String> getIncludes( String type )
336         throws MojoFailureException, MojoExecutionException
337     {
338         Set<String> result = new HashSet<String>();
339         if ( !type.equals( TEST ) && !includes.isEmpty() )
340         {
341             result.addAll( includes );
342         }
343         else if ( type.equals( TEST ) && !testIncludes.isEmpty() )
344         {
345             result.addAll( testIncludes );
346         }
347         else
348         {
349             String defaultIncludes = NarProperties.getInstance(mojo.getMavenProject()).getProperty( getPrefix() + "includes" );
350             if ( defaultIncludes != null )
351             {
352                 String[] include = defaultIncludes.split( " " );
353                 for ( int i = 0; i < include.length; i++ )
354                 {
355                     result.add( include[i].trim() );
356                 }
357             }
358         }
359         return result;
360     }

421     public final CompilerDef getCompiler( String type, String output )
422         throws MojoFailureException, MojoExecutionException
423     {
424         String name = getName();
425         if (name == null) return null;
426         
427         CompilerDef compiler = new CompilerDef();
428         compiler.setProject( mojo.getAntProject() );
429         CompilerEnum compilerName = new CompilerEnum();
430         compilerName.setValue( name );
431         compiler.setName( compilerName );
432 
433         // tool path
434         if ( toolPath != null )
435         {
436             compiler.setToolPath( toolPath );
437         }
438 
439         // debug, exceptions, rtti, multiThreaded
440         compiler.setDebug( debug );
441         compiler.setExceptions( exceptions );
442         compiler.setRtti( rtti );
443         compiler.setMultithreaded( mojo.getOS().equals( "Windows" ) ? true : multiThreaded );
444 
445         // optimize
446         OptimizationEnum optimization = new OptimizationEnum();
447         optimization.setValue( optimize );
448         compiler.setOptimize( optimization );
449 
450         // add options
451         if ( options != null )
452         {
453             for ( Iterator<String> i = options.iterator(); i.hasNext(); )
454             {
455                 CompilerArgument arg = new CompilerArgument();
456                 arg.setValue( (String) i.next() );
457                 compiler.addConfiguredCompilerArg( arg );
458             }
459         }
460 
461         if ( optionSet != null )
462         {
463 
464             String[] opts = optionSet.split( "\\s" );
465 
466             for ( int i = 0; i < opts.length; i++ )
467             {
468 
469                 CompilerArgument arg = new CompilerArgument();
470 
471                 arg.setValue( opts[i] );
472                 compiler.addConfiguredCompilerArg( arg );
473             }
474         }
475 
476         compiler.setClearDefaultOptions(clearDefaultOptions);
477         if ( !clearDefaultOptions )
478         {
479             String optionsProperty = NarProperties.getInstance(mojo.getMavenProject()).getProperty( getPrefix() + "options" );
480             if ( optionsProperty != null )
481             {
482                 String[] option = optionsProperty.split( " " );
483                 for ( int i = 0; i < option.length; i++ )
484                 {
485                     CompilerArgument arg = new CompilerArgument();
486                     arg.setValue( option[i] );
487                     compiler.addConfiguredCompilerArg( arg );
488                 }
489             }
490         }
491 
492         // add defines
493         if ( defines != null )
494         {
495             DefineSet ds = new DefineSet();
496             for ( Iterator<String> i = defines.iterator(); i.hasNext(); )
497             {
498                 DefineArgument define = new DefineArgument();
499                 String[] pair = i.next().split( "=", 2 );
500                 define.setName( pair[0] );
501                 define.setValue( pair.length > 1 ? pair[1] : null );
502                 ds.addDefine( define );
503             }
504             compiler.addConfiguredDefineset( ds );
505         }
506 
507         if ( defineSet != null )
508         {
509 
510             String[] defList = defineSet.split( "," );
511             DefineSet defSet = new DefineSet();
512 
513             for ( int i = 0; i < defList.length; i++ )
514             {
515 
516                 String[] pair = defList[i].trim().split( "=", 2 );
517                 DefineArgument def = new DefineArgument();
518 
519                 def.setName( pair[0] );
520                 def.setValue( pair.length > 1 ? pair[1] : null );
521 
522                 defSet.addDefine( def );
523             }
524 
525             compiler.addConfiguredDefineset( defSet );
526         }
527 
528         if ( !clearDefaultDefines )
529         {
530             DefineSet ds = new DefineSet();
531             String defaultDefines = NarProperties.getInstance(mojo.getMavenProject()).getProperty( getPrefix() + "defines" );
532             if ( defaultDefines != null )
533             {
534                 ds.setDefine( new CUtil.StringArrayBuilder( defaultDefines ) );
535             }
536             compiler.addConfiguredDefineset( ds );
537         }
538 
539         // add undefines
540         if ( undefines != null )
541         {
542             DefineSet us = new DefineSet();
543             for ( Iterator<String> i = undefines.iterator(); i.hasNext(); )
544             {
545                 DefineArgument undefine = new DefineArgument();
546                 String[] pair = i.next().split( "=", 2 );
547                 undefine.setName( pair[0] );
548                 undefine.setValue( pair.length > 1 ? pair[1] : null );
549                 us.addUndefine( undefine );
550             }
551             compiler.addConfiguredDefineset( us );
552         }
553 
554         if ( undefineSet != null )
555         {
556 
557             String[] undefList = undefineSet.split( "," );
558             DefineSet undefSet = new DefineSet();
559 
560             for ( int i = 0; i < undefList.length; i++ )
561             {
562 
563                 String[] pair = undefList[i].trim().split( "=", 2 );
564                 DefineArgument undef = new DefineArgument();
565 
566                 undef.setName( pair[0] );
567                 undef.setValue( pair.length > 1 ? pair[1] : null );
568 
569                 undefSet.addUndefine( undef );
570             }
571 
572             compiler.addConfiguredDefineset( undefSet );
573         }
574 
575         if ( !clearDefaultUndefines )
576         {
577             DefineSet us = new DefineSet();
578             String defaultUndefines = NarProperties.getInstance(mojo.getMavenProject()).getProperty( getPrefix() + "undefines" );
579             if ( defaultUndefines != null )
580             {
581                 us.setUndefine( new CUtil.StringArrayBuilder( defaultUndefines ) );
582             }
583             compiler.addConfiguredDefineset( us );
584         }
585 
586         // add include path
587         for ( Iterator<IncludePath> i = getIncludePaths( type ).iterator(); i.hasNext(); )
588         {
589             IncludePath includePath = i.next();
590             // Darren Sargent, 30Jan2008 - fail build if invalid include path(s) specified.
591                         if ( ! includePath.exists() ) {
592                                 throw new MojoFailureException("NAR: Include path not found: " + includePath);
593                         }
594             compiler.createIncludePath().setPath( includePath.getPath() );
595         }
596 
597         // add system include path (at the end)
598         if ( systemIncludePaths != null )
599         {
600             for ( Iterator<String> i = systemIncludePaths.iterator(); i.hasNext(); )
601             {
602                 String path = i.next();
603                 compiler.createSysIncludePath().setPath( path );
604             }
605         }
606 
607         // Add default fileset (if exists)
608         List<File> srcDirs = getSourceDirectories( type );
609         Set<String> includeSet = getIncludes( type );
610         Set<String> excludeSet = getExcludes( type );
611 
612         // now add all but the current test to the excludes
613         for ( Iterator i = mojo.getTests().iterator(); i.hasNext(); )
614         {
615             Test test = (Test) i.next();
616             if ( !test.getName().equals( output ) )
617             {
618                 excludeSet.add( "**/" + test.getName() + ".*" );
619             }
620         }
621 
622         for ( Iterator<File> i = srcDirs.iterator(); i.hasNext(); )
623         {
624             File srcDir = i.next();
625             mojo.getLog().debug( "Checking for existence of " + getLanguage() + " source directory: " + srcDir );
626             if ( srcDir.exists() )
627             {
628                 if ( compileOrder != null )
629                 {
630                     compiler.setOrder( Arrays.asList( StringUtils.split( compileOrder, ", " ) ) );
631                 }
632 
633                 ConditionalFileSet fileSet = new ConditionalFileSet();
634                 fileSet.setProject( mojo.getAntProject() );
635                 fileSet.setIncludes( StringUtils.join( includeSet.iterator(), "," ) );
636                 fileSet.setExcludes( StringUtils.join( excludeSet.iterator(), "," ) );
637                 fileSet.setDir( srcDir );
638                 compiler.addFileset( fileSet );
639             }
640         }
641 
642         return compiler;
643     }
