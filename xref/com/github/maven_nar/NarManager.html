70    public final void downloadAttachedNars(final List/* <NarArtifacts> */narArtifacts, final List remoteRepositories,
71        final ArtifactResolver resolver, final String classifier) throws MojoExecutionException, MojoFailureException {
72      // FIXME this may not be the right way to do this.... -U ignored and
73      // also SNAPSHOT not used
74      final List dependencies = getAttachedNarDependencies(narArtifacts, classifier);
75  
76      this.log.debug("Download called with classifier: " + classifier + " for NarDependencies {");
77      for (final Iterator i = dependencies.iterator(); i.hasNext();) {
78        this.log.debug("  - " + i.next());
79      }
80      this.log.debug("}");
81  
82      for (final Iterator i = dependencies.iterator(); i.hasNext();) {
83        final Artifact dependency = (Artifact) i.next();
84        try {
85          this.log.debug("Resolving " + dependency);
86          resolver.resolve(dependency, remoteRepositories, this.repository);
87        } catch (final ArtifactNotFoundException e) {
88          final String message = "nar not found " + dependency.getId();
89          throw new MojoExecutionException(message, e);
90        } catch (final ArtifactResolutionException e) {
91          final String message = "nar cannot resolve " + dependency.getId();
92          throw new MojoExecutionException(message, e);
93        }
94      }
95    }

97    private List/* <AttachedNarArtifact> */getAttachedNarDependencies(final Artifact dependency, final AOL archOsLinker,
98        final String type) throws MojoExecutionException, MojoFailureException {
99      AOL aol = archOsLinker;
100     this.log.debug("GetNarDependencies for " + dependency + ", aol: " + aol + ", type: " + type);
101     final List artifactList = new ArrayList();
102     final NarInfo narInfo = getNarInfo(dependency);
103     final String[] nars = narInfo.getAttachedNars(aol, type);
104     // FIXME Move this to NarInfo....
105     if (nars != null) {
106       for (final String nar2 : nars) {
107         this.log.debug("    Checking: " + nar2);
108         if (nar2.equals("")) {
109           continue;
110         }
111         final String[] nar = nar2.split(":", 5);
112         if (nar.length >= 4) {
113           try {
114             final String groupId = nar[0].trim();
115             final String artifactId = nar[1].trim();
116             final String ext = nar[2].trim();
117             String classifier = nar[3].trim();
118             // translate for instance g++ to gcc...
119             aol = narInfo.getAOL(aol);
120             if (aol != null) {
121               classifier = NarUtil.replace("${aol}", aol.toString(), classifier);
122             }
123             final String version = nar.length >= 5 ? nar[4].trim() : dependency.getBaseVersion();
124             artifactList.add(new AttachedNarArtifact(groupId, artifactId, version, dependency.getScope(), ext,
125                 classifier, dependency.isOptional(), dependency.getFile()));
126           } catch (final InvalidVersionSpecificationException e) {
127             throw new MojoExecutionException("Error while reading nar file for dependency " + dependency, e);
128           }
129         } else {
130           this.log.warn("nars property in " + dependency.getArtifactId() + " contains invalid field: '" + nar2
131               + "' for type: " + type);
132         }
133       }
134     }
135     return artifactList;
136   }

147   /**
148    * Returns a list of all attached nar dependencies for a specific binding and
149    * "noarch", but not where "local" is
150    * specified
151    * 
152    * @param scope
153    *          compile, test, runtime, ....
154    * @param aol
155    *          either a valid aol, noarch or null. In case of null both the
156    *          default getAOL() and noarch dependencies
157    *          are returned.
158    * @param type
159    *          noarch, static, shared, jni, or null. In case of null the default
160    *          binding found in narInfo is used.
161    * @return
162    * @throws MojoExecutionException
163    * @throws MojoFailureException
164    */
165   public final List/* <AttachedNarArtifact> */getAttachedNarDependencies(final List/*
166                                                                                     * <
167                                                                                     * NarArtifacts
168                                                                                     * >
169                                                                                     */narArtifacts,
170       final AOL archOsLinker, final String type) throws MojoExecutionException, MojoFailureException {
171     boolean noarch = false;
172     AOL aol = archOsLinker;
173     if (aol == null) {
174       noarch = true;
175       aol = this.defaultAOL;
176     }
177 
178     final List artifactList = new ArrayList();
179     for (final Iterator i = narArtifacts.iterator(); i.hasNext();) {
180       final Artifact dependency = (Artifact) i.next();
181       final NarInfo narInfo = getNarInfo(dependency);
182       if (noarch) {
183         artifactList.addAll(getAttachedNarDependencies(dependency, null, NarConstants.NAR_NO_ARCH));
184       }
185 
186       // use preferred binding, unless non existing.
187       final String binding = narInfo.getBinding(aol, type != null ? type : Library.STATIC);
188 
189       // FIXME kludge, but does not work anymore since AOL is now a class
190       if (aol.equals(NarConstants.NAR_NO_ARCH)) {
191         // FIXME no handling of local
192         artifactList.addAll(getAttachedNarDependencies(dependency, null, NarConstants.NAR_NO_ARCH));
193       } else {
194         artifactList.addAll(getAttachedNarDependencies(dependency, aol, binding));
195       }
196     }
197     return artifactList;
198   }

200   public final List/* <AttachedNarArtifact> */getAttachedNarDependencies(final List/*
201                                                                                     * <
202                                                                                     * NarArtifacts
203                                                                                     * >
204                                                                                     */narArtifacts,
205       final String classifier) throws MojoExecutionException, MojoFailureException {
206     AOL aol = null;
207     String type = null;
208     if (classifier != null) {
209       final int dash = classifier.lastIndexOf('-');
210       if (dash < 0) {
211         aol = new AOL(classifier);
212       } else {
213         aol = new AOL(classifier.substring(0, dash));
214         type = classifier.substring(dash + 1);
215       }
216     }
217     return getAttachedNarDependencies(narArtifacts, aol, type);
218   }

249   public final NarInfo getNarInfo(final Artifact dependency) throws MojoExecutionException {
250     // FIXME reported to maven developer list, isSnapshot changes behaviour
251     // of getBaseVersion, called in pathOf.
252     dependency.isSnapshot();
253 
254     final File file = new File(this.repository.getBasedir(), this.repository.pathOf(dependency));
255     if (!file.exists()) {
256       return null;
257     }
258 
259     JarFile jar = null;
260     try {
261       jar = new JarFile(file);
262       final NarInfo info = new NarInfo(dependency.getGroupId(), dependency.getArtifactId(),
263           dependency.getBaseVersion(), this.log);
264       if (!info.exists(jar)) {
265         return null;
266       }
267       info.read(jar);
268       return info;
269     } catch (final IOException e) {
270       throw new MojoExecutionException("Error while reading " + file, e);
271     } finally {
272       if (jar != null) {
273         try {
274           jar.close();
275         } catch (final IOException e) {
276           // ignore
277         }
278       }
279     }
280   }

282   public final void unpackAttachedNars(final List/* <NarArtifacts> */narArtifacts,
283       final ArchiverManager archiverManager, final String classifier, final String os, final NarLayout layout,
284       final File unpackDir) throws MojoExecutionException, MojoFailureException {
285     this.log.debug("Unpack called for OS: " + os + ", classifier: " + classifier + " for NarArtifacts {");
286     for (final Iterator i = narArtifacts.iterator(); i.hasNext();) {
287       this.log.debug("  - " + i.next());
288     }
289     this.log.debug("}");
290     // FIXME, kludge to get to download the -noarch, based on classifier
291     final List dependencies = getAttachedNarDependencies(narArtifacts, classifier);
292     for (final Iterator i = dependencies.iterator(); i.hasNext();) {
293       final Artifact dependency = (Artifact) i.next();
294       this.log.debug("Unpack " + dependency + " to " + unpackDir);
295       final File file = getNarFile(dependency);
296 
297       layout.unpackNar(unpackDir, archiverManager, file, os, this.linkerName, this.defaultAOL);
298     }
